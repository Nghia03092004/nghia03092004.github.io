\section{Problem Statement}
For $a_i = 153^i \bmod 10\,000\,019$, let $S(n)$ be the sum of all elements in all length-4 ascending subsequences of $(a_1, \ldots, a_n)$. Find $S(10^6) \bmod 10^9+7$.

\section{Fenwick Tree Decomposition}

\begin{theorem}
Define $c_k(j)$ = count of ascending subsequences of length $k$ ending at index $j$, and $s_k(j)$ = their element sum. Then:
\[
c_k(j) = \sum_{\substack{i<j\\a_i<a_j}} c_{k-1}(i), \qquad s_k(j) = \sum_{\substack{i<j\\a_i<a_j}} s_{k-1}(i) + a_j \cdot c_k(j).
\]
These prefix sums are computable in $O(\log n)$ per query using Fenwick trees on coordinate-compressed values.
\end{theorem}

\section{Verification}
$S(6) = 94\,513\,710$, $S(100) = 4\,465\,488\,724\,217$.

\section{Complexity}
$O(n \log n)$ time, $O(n)$ space.
