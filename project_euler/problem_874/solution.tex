\section{Problem Statement}

Given $\{1,\ldots,N\}$, find the maximum-size pairwise coprime subset:
\[
\max |S| \quad \text{s.t.\ } \gcd(a,b)=1 \;\forall\, a \neq b \in S.
\]

\section{Mathematical Foundation}

\begin{definition}[Squarefree Kernel]
$\operatorname{rad}(n) = \prod_{p \mid n} p$.
\end{definition}

\begin{lemma}[Coprimality via Radicals]
$\gcd(a,b) = 1$ iff $\gcd(\operatorname{rad}(a), \operatorname{rad}(b)) = 1$.
\end{lemma}

\begin{proof}
$\gcd(a,b)>1$ iff some prime $p$ divides both $a$ and $b$, iff $p \mid \operatorname{rad}(a)$ and $p \mid \operatorname{rad}(b)$.
\end{proof}

\begin{theorem}[Maximum Pairwise Coprime Set]
The maximum size of a pairwise coprime subset of $\{1,\ldots,N\}$ is $1 + \pi(N)$.
\end{theorem}

\begin{proof}
\emph{Lower bound:} $\{1\} \cup \{p \le N : p \text{ prime}\}$ is pairwise coprime with size $1 + \pi(N)$.

\emph{Upper bound:} For $m > 1$, let $p(m)$ be its smallest prime factor. If $a,b \in S \setminus\{1\}$ share smallest prime factor $p$, then $p \mid \gcd(a,b)$. So $m \mapsto p(m)$ is injective on $S \setminus\{1\}$, giving $|S\setminus\{1\}| \le \pi(N)$.
\end{proof}

\begin{theorem}[Coprime Pair Density]
$\Pr[\gcd(a,b)=1] \to 6/\pi^2$ as $N \to \infty$.
\end{theorem}

\begin{proof}
By inclusion-exclusion over primes:
\[
\Pr[\gcd(a,b)=1] = \prod_{p} \left(1 - \frac{1}{p^2}\right) = \frac{1}{\zeta(2)} = \frac{6}{\pi^2}. \qedhere
\]
\end{proof}

\begin{lemma}[NP-hardness of Weighted Variant]
Maximum-weight pairwise coprime subset with arbitrary weights is NP-hard.
\end{lemma}

\begin{proof}
Reduces from maximum weighted independent set on the non-coprimality graph.
\end{proof}

\section{Algorithm}

For the unweighted problem: sieve primes up to $N$ and take $\{1\} \cup \{\text{primes}\}$. For the weighted variant: DP over prime factor subsets.

\section{Complexity Analysis}

\begin{itemize}
\item \textbf{Time:} $O(N \log\log N)$ for the sieve; $O(N \cdot 2^{\omega})$ for weighted DP where $\omega$ = max distinct prime factors.
\item \textbf{Space:} $O(N)$ for sieve; $O(2^{\omega})$ for DP states.
\end{itemize}
