\begin{theorem}
For coprime $h,k$: $s(h,k) + s(k-h,k) = 0$.
\end{theorem}
\begin{proof}
Since $((x)) + ((-x)) = 0$ for non-integer $x$, the terms pair up under $h \to k-h$.
\end{proof}
