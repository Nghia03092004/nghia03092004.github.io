\section{Problem Statement}
Let $R(a, b, c)$ be the maximum area covered by three non-overlapping circles inside a triangle with edge lengths $a$, $b$, and $c$.

Let $S(n)$ be the average value of $R(a, b, c)$ over all integer triplets $(a, b, c)$ such that $1 \le a \le b \le c < a + b \le n$.

Given: $S(2) = R(1,1,1) \approx 0.31998$, $S(5) \approx 1.25899$.

Find $S(1803)$ rounded to 5 decimal places.

\section{Mathematical Analysis}

\subsection{Malfatti Circles}
For a triangle with angles $A$, $B$, $C$, semiperimeter $s$, inradius $r$, and area $\Delta$, the Malfatti circle radii are:
\[
r_1 = \frac{r}{2(1 + \tan(A/2))} \cdot \sec^2(A/2), \quad \text{etc.}
\]

More precisely, using Steiner's formulas:
\[
r_1 = \frac{r}{2} \left(1 - \frac{r}{s - a}\right) \sec^2\!\left(\frac{A}{2}\right)
\]

\subsection{Greedy Packing}
The greedy algorithm:
\begin{enumerate}
    \item Inscribe the largest circle in the triangle (the incircle with radius $r = \Delta/s$).
    \item Find the largest circle fitting in each of the three ``corner'' regions.
    \item Select the two largest of these corner circles.
\end{enumerate}

The corner circle opposite vertex $A$ has approximate radius:
\[
r_A = r \cdot \frac{\tan(A/2)}{1 + \tan(A/2)}
\]

\subsection{Optimal Packing}
For each triangle, we compute both the Malfatti circle areas and the greedy packing areas, then take the maximum:
\[
R(a,b,c) = \max\left(\pi(r_1^2 + r_2^2 + r_3^2)_{\text{Malfatti}},\; \pi(r^2 + r_{c1}^2 + r_{c2}^2)_{\text{greedy}}\right)
\]

\subsection{Averaging}
\[
S(n) = \frac{1}{N} \sum_{\substack{1 \le a \le b \le c \\ c < a+b \le n}} R(a,b,c)
\]
where $N$ is the number of valid triplets.

\section{Answer}

\[
\boxed{110242.87794}
\]
