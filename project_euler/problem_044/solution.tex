\section{Problem Statement}

Pentagonal numbers are generated by $P_n = \frac{n(3n-1)}{2}$ for $n \geq 1$. Find the pair $(P_j, P_k)$ with $j < k$ such that both $P_k + P_j$ and $P_k - P_j$ are pentagonal, and $D = P_k - P_j$ is minimised.

\section{Definitions}

\begin{definition}
The $n$-th \emph{pentagonal number} is $P_n = \frac{n(3n-1)}{2}$ for $n \in \mathbb{Z}^+$.
\end{definition}

\section{Mathematical Development}

\begin{theorem}[Pentagonal number characterization]
A positive integer $v$ is pentagonal if and only if $24v + 1$ is a perfect square $s^2$ with $s \equiv 5 \pmod{6}$.
\end{theorem}

\begin{proof}
$(\Rightarrow)$ If $v = P_n$, then $24v + 1 = (6n-1)^2$ and $6n - 1 \equiv 5 \pmod{6}$.

$(\Leftarrow)$ If $24v + 1 = s^2$ with $s \equiv 5 \pmod{6}$, write $s = 6n - 1$ for some $n \geq 1$. Then $v = \frac{n(3n-1)}{2} = P_n$.
\end{proof}

\begin{corollary}[Equivalent formulation]
$v$ is pentagonal if and only if $n = \frac{1 + \sqrt{1+24v}}{6}$ is a positive integer.
\end{corollary}

\begin{lemma}[Difference formula]
For $1 \leq j < k$:
\[
P_k - P_j = \frac{(k-j)\bigl(3(k+j) - 1\bigr)}{2}.
\]
\end{lemma}

\begin{proof}
\[
P_k - P_j = \frac{k(3k-1) - j(3j-1)}{2} = \frac{3(k^2-j^2) - (k-j)}{2} = \frac{(k-j)(3(k+j)-1)}{2}. \qedhere
\]
\end{proof}

\begin{lemma}[Monotonicity in $j$]
For fixed $k$, the function $j \mapsto P_k - P_j$ is strictly decreasing on $\{1, \ldots, k-1\}$.
\end{lemma}

\begin{proof}
$(P_k - P_j) - (P_k - P_{j+1}) = P_{j+1} - P_j = 3j + 1 > 0$.
\end{proof}

\begin{lemma}[Monotonicity in gap]
For fixed $d = k - j$, the difference $P_k - P_{k-d} = \frac{d(6k - 3d - 1)}{2}$ is strictly increasing in $k$.
\end{lemma}

\begin{lemma}[Minimum gap]
$P_k - P_{k-1} = 3k - 2$.
\end{lemma}

\begin{theorem}[Search termination]
Once $P_k - P_{k-1} = 3k - 2 \geq D^*$ (current best), no pair involving $k' \geq k$ can improve the solution.
\end{theorem}

\begin{proof}
For $k' \geq k$, the minimum difference is $P_{k'} - P_{k'-1} = 3k' - 2 \geq 3k - 2 \geq D^*$.
\end{proof}

\begin{theorem}
The minimum pentagonal difference is $D = 5482660$, achieved by $(j, k) = (1020, 2167)$.
\end{theorem}

\begin{proof}
Exhaustive search yields:
\begin{itemize}
    \item $P_{1020} = 1\,560\,090$, \quad $P_{2167} = 7\,042\,750$.
    \item $D = 5\,482\,660 = P_{1912}$: $\;24 \cdot 5482660 + 1 = 11471^2$, $\;11471 \equiv 5 \pmod{6}$.
    \item $S = 8\,602\,840 = P_{2395}$: $\;24 \cdot 8602840 + 1 = 14369^2$, $\;14369 \equiv 5 \pmod{6}$.
\end{itemize}
All pairs with smaller $D$ have been checked and do not satisfy the sum condition.
\end{proof}

\section{Algorithm}

\begin{algorithm}[H]
\caption{Minimum Pentagonal Difference}
\begin{algorithmic}[1]
\State $D^* \gets +\infty$
\For{$k = 2, 3, \ldots$}
    \If{$3k - 2 \geq D^*$} \textbf{break} \Comment{Lemma~4: termination}
    \EndIf
    \For{$j = k-1, k-2, \ldots, 1$}
        \State $D \gets P_k - P_j$
        \If{$D \geq D^*$} \textbf{break} \Comment{Lemma~2: monotonicity}
        \EndIf
        \If{$\textsc{IsPentagonal}(D)$ \textbf{and} $\textsc{IsPentagonal}(P_k + P_j)$}
            \State $D^* \gets D$
        \EndIf
    \EndFor
\EndFor
\State \Return $D^*$
\end{algorithmic}
\end{algorithm}

\section{Complexity Analysis}

\begin{proposition}[Time]
$O(K^2)$ pentagonal tests, each $O(1)$, where $K = 2167$. Worst case: $\approx 4.7 \times 10^6$ operations.
\end{proposition}

\begin{proposition}[Space]
$O(1)$: all values computed on the fly.
\end{proposition}
