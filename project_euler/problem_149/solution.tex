\section{Problem Statement}
Find the greatest sum of any contiguous subsequence in any direction (horizontal, vertical, diagonal, anti-diagonal) in a $2000 \times 2000$ table generated by a lagged Fibonacci generator.

\section{Generator}
The sequence $s_k$ for $k = 1, \ldots, 4{,}000{,}000$:
\[
s_k = \begin{cases}
(100003 - 200003k + 300007k^3) \bmod 10^6 - 500000 & \text{if } 1 \le k \le 55 \\
(s_{k-24} + s_{k-55} + 10^6) \bmod 10^6 - 500000 & \text{if } k > 55
\end{cases}
\]
Entry at row $i$, column $j$ (1-indexed): $s_{2000(i-1)+j}$.

\section{Algorithm}

\subsection{Kadane's Algorithm}
For a sequence $(a_1, \ldots, a_n)$, the maximum contiguous subsequence sum is found in $O(n)$:
\begin{algorithmic}
\State $\text{maxHere} \gets 0$
\State $\text{maxSoFar} \gets -\infty$
\For{$i = 1$ to $n$}
    \State $\text{maxHere} \gets \max(a_i,\, \text{maxHere} + a_i)$
    \State $\text{maxSoFar} \gets \max(\text{maxSoFar},\, \text{maxHere})$
\EndFor
\State \Return $\text{maxSoFar}$
\end{algorithmic}

\subsection{Four Directions}
Apply Kadane's algorithm along:
\begin{enumerate}
    \item \textbf{Horizontal}: 2000 rows.
    \item \textbf{Vertical}: 2000 columns.
    \item \textbf{Diagonal} ($\searrow$): $2 \times 2000 - 1 = 3999$ diagonals.
    \item \textbf{Anti-diagonal} ($\swarrow$): $3999$ anti-diagonals.
\end{enumerate}

The overall answer is the maximum across all directions.

\section{Complexity}
\begin{itemize}
    \item Table generation: $O(N^2)$.
    \item Kadane's over all directions: $O(N^2)$ per direction.
    \item Total: $O(N^2)$ with $N = 2000$.
\end{itemize}
