\section{Problem Statement}
A permutation of $\{1, 2, \ldots, 100\}$ is chosen uniformly at random. What is the probability that exactly 22 of the 25 primes among $1$--$100$ are not in their natural position (equivalently, exactly 3 primes are fixed points)? Round to 12 decimal places.

\section{Mathematical Foundation}

\begin{theorem}[Partial Derangement with Unrestricted Elements]
The number of permutations of $\{1,\ldots,100\}$ in which exactly 3 of the 25 primes are fixed points is
\[
N = \binom{25}{3} \sum_{k=0}^{22} (-1)^k \binom{22}{k}(97-k)!\,.
\]
\end{theorem}

\begin{proof}
Choose which 3 primes are fixed: $\binom{25}{3}$ ways. The remaining 97 elements fill 97 positions, with 22 displaced primes each having one forbidden position. By inclusion-exclusion over the 22 forbidden constraints:
\[
\left|\bigcap_{i=1}^{22} \overline{F_i}\right| = \sum_{k=0}^{22} (-1)^k \binom{22}{k}(97-k)!
\]
since fixing $k$ additional primes to their forbidden positions leaves $(97-k)!$ free permutations, and there are $\binom{22}{k}$ ways to choose which $k$ are additionally fixed.
\end{proof}

\begin{lemma}[Probability]
\[
P = \frac{N}{100!} = \frac{2300}{100!} \sum_{k=0}^{22} (-1)^k \binom{22}{k}(97-k)!
\]
Equivalently, factoring out $75!$:
\[
P = \frac{2300 \cdot 75!}{100!} \sum_{k=0}^{22} (-1)^k \binom{22}{k} \frac{(97-k)!}{75!}
\]
where $(97-k)!/75!$ is the falling factorial $(97-k)^{\underline{22-k}}$.
\end{lemma}

\begin{proof}
Direct substitution of $N$ into $P = N/100!$ and $\binom{25}{3} = 2300$.
\end{proof}

\section{Algorithm}

\begin{verbatim}
function solve():
    total = 0
    for k = 0 to 22:
        total += (-1)^k * C(22,k) * (97-k)!
    return 2300 * total / 100!
\end{verbatim}

\section{Complexity Analysis}
\begin{itemize}
    \item \textbf{Time:} $O(n)$ where $n = 22$ inclusion-exclusion terms.
    \item \textbf{Space:} $O(1)$.
\end{itemize}

\section{Answer}

\[
\boxed{0.001887854841}
\]
