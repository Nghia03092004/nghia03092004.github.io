\section{Problem Statement}

How many ways can 60 black objects and 40 white objects be grouped into non-empty groups, where the order of groups does not matter and each group is characterised by the pair $(b, w)$ denoting the number of black and white objects it contains?

\section{Mathematical Development}

\begin{definition}
A \emph{bicolour partition} of the pair $(B, W) \in \mathbb{Z}_{\ge 0}^2$ is a multiset $\{(b_1, w_1), \ldots, (b_k, w_k)\}$ of elements from $\mathbb{Z}_{\ge 0}^2 \setminus \{(0,0)\}$ satisfying $\sum_{i=1}^k b_i = B$ and $\sum_{i=1}^k w_i = W$. Let $P(B, W)$ denote the number of such partitions.
\end{definition}

\begin{theorem}[Generating Function]
The bicolour partition function satisfies
\[
\sum_{B, W \ge 0} P(B, W)\, x^B y^W = \prod_{\substack{(b, w) \in \mathbb{Z}_{\ge 0}^2 \\ (b,w) \neq (0,0)}} \frac{1}{1 - x^b y^w}.
\]
\end{theorem}

\begin{proof}
A bicolour partition is a function $\mu\colon \mathcal{P} \to \mathbb{Z}_{\ge 0}$ (where $\mathcal{P} = \mathbb{Z}_{\ge 0}^2 \setminus \{(0,0)\}$) with prescribed coordinate sums. The generating function for the multiplicity of a single part type $(b,w)$ is $\sum_{k \ge 0} (x^b y^w)^k = (1 - x^b y^w)^{-1}$. Independence of part types yields the product formula. The coefficient $[x^B y^W]$ counts precisely $P(B, W)$.
\end{proof}

\begin{theorem}[Unbounded Knapsack Reduction]
Enumerate part types as $p_1, \ldots, p_M$ in any fixed total order, $M = (B+1)(W+1)-1$. Define $f_j(b,w)$ as the number of bicolour partitions of $(b,w)$ using only parts from $\{p_1, \ldots, p_j\}$. Then $f_0(0,0)=1$, $f_0(b,w)=0$ for $(b,w) \neq (0,0)$, and for $p_j = (\delta_b, \delta_w)$:
\[
f_j(b, w) = \sum_{t=0}^{\lfloor \min(b/\delta_b,\, w/\delta_w) \rfloor} f_{j-1}(b - t\delta_b,\, w - t\delta_w).
\]
\end{theorem}

\begin{proof}
By the principle of sequential decision: for each part type $p_j$, choose multiplicity $t \ge 0$, then count partitions of the residual using preceding parts. The base case encodes the empty partition.
\end{proof}

\begin{corollary}[In-Place Computation]
Processing each part type $(\delta_b, \delta_w)$ with the update $\mathrm{dp}[b][w] \mathrel{+}= \mathrm{dp}[b - \delta_b][w - \delta_w]$ (indices increasing) computes $P(B, W)$ in place.
\end{corollary}

\begin{proof}
Standard unbounded knapsack argument. Increasing-order iteration permits unlimited multiplicity. By induction on the number of processed part types, $\mathrm{dp}[b][w] = f_j(b,w)$ after processing $p_j$.
\end{proof}

\section{Algorithm}

\begin{algorithmic}[1]
\Function{CountBicolourPartitions}{$B, W$}
    \State $\mathrm{dp}[0 \ldots B][0 \ldots W] \gets 0$;\quad $\mathrm{dp}[0][0] \gets 1$
    \For{$\delta_b = 0$ to $B$}
        \For{$\delta_w = 0$ to $W$}
            \If{$\delta_b = 0$ and $\delta_w = 0$} \textbf{continue} \EndIf
            \For{$b = \delta_b$ to $B$}
                \For{$w = \delta_w$ to $W$}
                    \State $\mathrm{dp}[b][w] \mathrel{+}= \mathrm{dp}[b - \delta_b][w - \delta_w]$
                \EndFor
            \EndFor
        \EndFor
    \EndFor
    \State \Return $\mathrm{dp}[B][W]$
\EndFunction
\end{algorithmic}

\section{Complexity Analysis}

\begin{itemize}
    \item \textbf{Time:} $O(B^2 W^2)$. There are $(B+1)(W+1)-1$ part types, each requiring at most $O(BW)$ work.
    \item \textbf{Space:} $O(BW)$ for the two-dimensional DP table.
\end{itemize}
