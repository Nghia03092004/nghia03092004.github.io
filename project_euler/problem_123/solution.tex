\section{Problem Statement}

Let $p_n$ be the $n$-th prime. Find the least $n$ such that
\[
r_n = \bigl[(p_n - 1)^n + (p_n + 1)^n\bigr] \bmod p_n^2 > 10^{10}.
\]

\section{Mathematical Development}

\begin{theorem}[Closed-form remainder]
Let $p = p_n$. Then:
\begin{itemize}
\item If $n$ is even: $r_n = 2$.
\item If $n$ is odd: $r_n \equiv 2np \pmod{p^2}$.
\end{itemize}
\end{theorem}

\begin{proof}
By the binomial theorem modulo $p^2$:
\begin{align*}
(p+1)^n &\equiv 1 + np \pmod{p^2}, \\
(p-1)^n &\equiv (-1)^n + np(-1)^{n-1} \pmod{p^2}.
\end{align*}
For even $n$: sum $= (1 - np) + (1 + np) = 2$. For odd $n$: sum $= (-1 + np) + (1 + np) = 2np$.
\end{proof}

\begin{lemma}[Exact remainder]
For odd $n \ge 5$, $2np_n < p_n^2$ (equivalently $2n < p_n$), so $r_n = 2np_n$ exactly.
\end{lemma}

\begin{proof}
By the Prime Number Theorem, $p_n \sim n \ln n > 2n$ for sufficiently large $n$. Direct verification confirms $p_n > 2n$ for all $n \ge 5$.
\end{proof}

\begin{theorem}[Characterization]
The answer is the smallest odd $n$ with $2np_n > 10^{10}$.
\end{theorem}

\begin{proof}
For even $n$, $r_n = 2$. For odd $n$ in the valid range, $r_n = 2np_n$ is strictly increasing, so the first exceedance determines the answer.
\end{proof}

\section{Algorithm}

\begin{enumerate}
\item Sieve primes up to $P = 300{,}000$.
\item For odd $n = 1, 3, 5, \ldots$: if $2np_n > 10^{10}$, return $n$.
\end{enumerate}

\section{Complexity Analysis}

\begin{itemize}
\item \textbf{Time:} $O(P \log \log P)$ for the sieve; $O(n^*/2)$ for the linear scan where $n^* = 21{,}035$.
\item \textbf{Space:} $O(P)$ for the sieve.
\end{itemize}

\section{Answer}

\[
\boxed{21035}
\]
