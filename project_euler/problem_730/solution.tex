\section{Problem Statement}
A triple $(p,q,r)$ is a \textbf{$k$-shifted Pythagorean triple} if $p^2+q^2+k=r^2$ with $\gcd(p,q,r)=1$, $p \le q \le r$. Count $P_k(n)$ for $p+q+r \le n$, and compute $S(m,n) = \sum_{k=0}^{m} P_k(n)$.

\section{Factorization Method}

\begin{theorem}
$(r-q)(r+q) = p^2+k$. Each factorization $p^2+k = d_1 d_2$ with $d_1 \le d_2$, $d_1 \equiv d_2 \pmod{2}$ yields $q = (d_2-d_1)/2$, $r = (d_1+d_2)/2$.
\end{theorem}

\begin{proof}
$r^2 - q^2 = (r-q)(r+q) = p^2+k$. Setting $d_1 = r-q$, $d_2 = r+q$ gives the result. The parity condition ensures $q,r$ are integers.
\end{proof}
