\section{Problem}
Compute ${}^n a \bmod m$ (modular tetration).
\begin{theorem}
$a^b \equiv a^{b \bmod \varphi(m)} \pmod{m}$ when $\gcd(a,m)=1$.
\end{theorem}
Recursion terminates in $O(\log m)$ steps since $\varphi^{(k)}(m) = 1$ after $\le \log_2 m$ iterations.
\section{Answer}
