\section{Problem Statement}

Find the number of non-isomorphic simple graphs on 7~vertices.

\section{Burnside's Lemma}

\begin{theorem}[Burnside/Cauchy-Frobenius]
Let a finite group~$G$ act on a finite set~$X$. The number of orbits is
\[
|X/G| = \frac{1}{|G|}\sum_{g \in G}|X^g|,
\]
where $X^g = \{x \in X : g \cdot x = x\}$ is the set of elements fixed by~$g$.
\end{theorem}

\begin{proof}
Count pairs $(g,x)$ with $g \cdot x = x$:
\[
\sum_{g \in G}|X^g| = \sum_{x \in X}|\mathrm{Stab}(x)| = \sum_{x \in X}\frac{|G|}{|\mathrm{Orb}(x)|} = |G|\cdot|\text{orbits}|. \qedhere
\]
\end{proof}

\section{Application to Graph Counting}

\begin{definition}
The symmetric group $S_n$ acts on the set of all $2^{\binom{n}{2}}$ simple graphs on $[n]$ by permuting vertices. Two graphs are isomorphic iff they are in the same orbit.
\end{definition}

\begin{theorem}
A permutation $\pi \in S_n$ fixes exactly $2^{c(\pi)}$ graphs, where $c(\pi)$ is the number of orbits of the induced action of $\pi$ on the edge set $\binom{[n]}{2}$.
\end{theorem}

\begin{proof}
$\pi$ permutes edges. A graph is fixed iff each edge orbit is entirely present or absent: $2^{c(\pi)}$ choices.
\end{proof}

\section{Edge Orbits from Cycle Type}

\begin{theorem}
For $\pi$ with cycle type $(\lambda_1, \ldots, \lambda_k)$:
\[
c(\pi) = \sum_{i<j}\gcd(\lambda_i, \lambda_j) + \sum_{i}\lfloor \lambda_i/2 \rfloor.
\]
\end{theorem}

\begin{proof}
Edges split into inter-cycle pairs (between cycles $C_i, C_j$, contributing $\gcd(\lambda_i, \lambda_j)$ orbits each) and intra-cycle pairs (within cycle $C_i$, contributing $\lfloor \lambda_i/2 \rfloor$ orbits).
\end{proof}

\section{Conjugacy Classes}

\begin{proposition}
Permutations with cycle type $(\lambda_1^{a_1}, \lambda_2^{a_2}, \ldots)$ number
\[
\frac{n!}{\prod_i \lambda_i^{a_i} \cdot a_i!}.
\]
\end{proposition}
