\section{Problem Statement}
Find $Q(10^{12})$, count of squarefree numbers up to $10^{12}$.

\section{Inclusion-Exclusion}
\begin{theorem}
$Q(n) = \sum_{d=1}^{\lfloor\sqrt{n}\rfloor} \mu(d) \lfloor n/d^2 \rfloor$.
\end{theorem}

\begin{proof}
$\sum_{d^2|m} \mu(d) = [m \text{ squarefree}]$ by M\"obius inversion. Sum over $m \leq n$.
\end{proof}

\section{Asymptotics}
\begin{theorem}
$Q(n) = 6n/\pi^2 + O(\sqrt{n})$.
\end{theorem}
