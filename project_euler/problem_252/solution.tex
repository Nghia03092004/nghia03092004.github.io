\section{Problem Statement}
Given 500 points generated by the LCG:
\begin{align}
s_0 &= 290797, \\
s_{n+1} &= s_n^2 \bmod 50515093, \\
t_n &= (s_n \bmod 2000) - 1000,
\end{align}
with points $P_k = (t_{2k}, t_{2k+1})$ for $k = 0, \ldots, 499$, find the area of the largest convex polygon whose vertices are from the given points and which contains no other given point in its interior. Report the answer rounded to one decimal place.

\section{Formal Mathematical Development}

\begin{definition}
Let $S \subset \mathbb{R}^2$ be finite. A \emph{convex hole} in $S$ is a convex polygon $P$ with $\mathrm{Vert}(P) \subseteq S$ and $\mathrm{int}(P) \cap S = \emptyset$.
\end{definition}

\begin{definition}[Signed Area]
For $A, B, C \in \mathbb{R}^2$:
\[
\sigma(A, B, C) = (B_x - A_x)(C_y - A_y) - (C_x - A_x)(B_y - A_y).
\]
\end{definition}

\begin{lemma}[Point-in-Triangle Test]
A point $Q$ lies strictly inside $\triangle P_1 P_2 P_3$ if and only if the three signed areas $\sigma(P_1, P_2, Q)$, $\sigma(P_2, P_3, Q)$, $\sigma(P_3, P_1, Q)$ are all strictly positive or all strictly negative.
\end{lemma}

\begin{proof}
The barycentric coordinates of $Q$ are $\lambda_i = \sigma_i / \sigma$. The point $Q$ is in the open interior iff all $\lambda_i > 0$, iff all $\sigma_i$ share the sign of $\sigma$.
\end{proof}

\begin{theorem}[Fan Triangulation Emptiness]
Let $P$ be a convex polygon with vertices $v_0, \ldots, v_{k-1}$ in CCW order. Then $\mathrm{int}(P) \cap S = \emptyset$ iff for every $i \in \{1, \ldots, k-2\}$, $\triangle v_0 v_i v_{i+1}$ contains no point of $S$ in its interior.
\end{theorem}

\begin{proof}
The fan $\{\triangle v_0 v_i v_{i+1}\}_{i=1}^{k-2}$ partitions $\mathrm{int}(P)$, so any $Q \in \mathrm{int}(P) \cap S$ lies in exactly one fan triangle.
\end{proof}

\begin{theorem}[Anchor-Based DP]
Fix an anchor $p_0 \in S$ and sort remaining points by polar angle. Define $\mathrm{dp}[j][k]$ as the maximum double-area of a convex polygon with anchor $p_0$, last two vertices $p_j, p_k$. Then:
\[
\mathrm{dp}[k][\ell] = \max_{j} \bigl(\mathrm{dp}[j][k] + \sigma(p_0, p_k, p_\ell)\bigr)
\]
subject to: left turn at $p_k$, empty fan triangle $\triangle p_0 p_k p_\ell$, and angular span feasibility.
\end{theorem}

\begin{proof}
Each extension appends $p_\ell$ to the polygon, adding fan triangle $\triangle p_0 p_k p_\ell$. By Theorem~2, emptiness of each fan triangle ensures emptiness of the entire polygon. Left-turn at $p_k$ maintains convexity. Tracking the first vertex enables closure checks.
\end{proof}

\section{Complexity}

\textbf{Time:} Precomputing bitsets: $O(n^3)$. DP per anchor: $O(n^3)$. Total: $O(n^4)$ with $n = 500$.

\textbf{Space:} $O(n^3/64)$ for bitsets; $O(n^2)$ for the DP table per anchor.
