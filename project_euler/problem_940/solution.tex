\section{Problem Statement}

Define $T(a, 1) = a$ and $T(a, n) = a^{T(a, n-1)}$ for $n \geq 2$. Find $T(2, 100) \bmod (10^9 + 7)$.

\section{Mathematical Foundation}

\begin{theorem}[Euler's theorem]
If $\gcd(a, m) = 1$, then $a^{\varphi(m)} \equiv 1 \pmod{m}$.
\end{theorem}

\begin{proof}
The map $r \mapsto ar$ is a bijection on $(\mathbb{Z}/m\mathbb{Z})^*$. Taking the product: $a^{\varphi(m)} \prod r \equiv \prod r \pmod{m}$. Cancel $\prod r$ (which is coprime to $m$).
\end{proof}

\begin{lemma}[Exponent reduction]
If $\gcd(a, m) = 1$ and $b \geq \log_2 m$, then $a^b \equiv a^{b \bmod \varphi(m)} \pmod{m}$.
\end{lemma}

\begin{proof}
Write $b = q \cdot \varphi(m) + r$. Then $a^b = (a^{\varphi(m)})^q \cdot a^r \equiv a^r \pmod{m}$.
\end{proof}

\begin{theorem}[Iterated totient reaches 1]
For $m \geq 2$: $\varphi^{(k)}(m) = 1$ for all $k \geq \lfloor \log_2 m \rfloor + 1$.
\end{theorem}

\begin{proof}
For $m \geq 3$: $\varphi(m) \leq m/2$ (the smallest prime factor $p \mid m$ satisfies $\varphi(m) \leq m(1 - 1/p) \leq m/2$). For $m = 2$: $\varphi(2) = 1$. By induction, $\varphi^{(k)}(m) \leq m/2^k$, so $\varphi^{(k)}(m) = 1$ once $k > \log_2 m$.
\end{proof}

\begin{theorem}[Tower convergence]
For $a \geq 2$ and $m \geq 2$, $T(a, n) \bmod m$ stabilizes for $n \geq \lfloor \log_2 m \rfloor + 1$. In particular, $T(2, 100) \equiv T(2, 31) \pmod{10^9 + 7}$.
\end{theorem}

\begin{proof}
The iterated totient chain starting from $m = 10^9 + 7$ reaches 1 in at most $\lfloor \log_2(10^9 + 7) \rfloor + 1 \leq 31$ steps. Once the modulus is 1, all residues are 0, so deeper recursion levels do not affect the result. Since $100 > 31$, the tower has converged.
\end{proof}

\begin{lemma}[Primality of $10^9 + 7$]
$p = 10^9 + 7$ is prime, so $\varphi(p) = p - 1 = 10^9 + 6$.
\end{lemma}

\begin{proof}
Verified by deterministic primality testing.
\end{proof}

\section{Algorithm}

Recursive function: to compute $T(a, n) \bmod m$, reduce the exponent modulo $\varphi(m)$ and recurse. Base cases: $m = 1 \Rightarrow 0$; $n = 1 \Rightarrow a \bmod m$. Recursion depth $\leq 31$.

\section{Complexity Analysis}

\begin{itemize}
\item \textbf{Time:} $O(\sqrt{m} \log m)$. Recursion depth $d = O(\log m)$; each level costs $O(\sqrt{m_i})$ for the totient and $O(\log m_i)$ for modular exponentiation.
\item \textbf{Space:} $O(\log m)$ for the recursion stack.
\end{itemize}
