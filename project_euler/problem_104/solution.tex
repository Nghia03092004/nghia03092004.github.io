\section{Problem Statement}

Find the smallest index $k$ such that the Fibonacci number $F_k$ has both its first 9 digits and its last 9 digits $1$--$9$ pandigital.

\section{Mathematical Development}

\begin{theorem}[Binet's Formula]
For all $n \geq 1$,
\[
F_n = \frac{\varphi^n - \psi^n}{\sqrt{5}}, \qquad \varphi = \frac{1+\sqrt{5}}{2}, \quad \psi = \frac{1-\sqrt{5}}{2}.
\]
\end{theorem}

\begin{proof}
Both $\varphi$ and $\psi$ satisfy $x^2 = x + 1$. Hence $f(n) = (\varphi^n - \psi^n)/\sqrt{5}$ satisfies $f(n) = f(n-1) + f(n-2)$ with $f(1) = f(2) = 1$. By induction, $f = F$.
\end{proof}

\begin{theorem}[Leading Digits]
\label{thm:leading}
Define $L_n = n\log_{10}\varphi - \tfrac{1}{2}\log_{10}5$. For $n$ sufficiently large, the first $m$ significant digits of $F_n$ equal $\lfloor 10^{\{L_n\}+m-1}\rfloor$, where $\{L_n\}$ denotes the fractional part.
\end{theorem}

\begin{proof}
Since $|\psi| < 1$, Binet gives $F_n \approx \varphi^n/\sqrt{5}$ with exponentially vanishing error. Taking $\log_{10}$: $\log_{10} F_n \approx L_n$. The fractional part of $L_n$ determines the significand, hence the leading digits. The approximation error is $O(|\psi|^n) < 10^{-14}$ for $n > 70$, far below the precision needed for 9 digits.
\end{proof}

\begin{lemma}[Modular Recurrence]
$F_n \bmod 10^9$ satisfies the same recurrence $F_n \equiv F_{n-1} + F_{n-2} \pmod{10^9}$.
\end{lemma}

\begin{proof}
Reduction modulo $10^9$ is a ring homomorphism preserving addition.
\end{proof}

\section{Algorithm}

\begin{enumerate}
\item Maintain $F_k \bmod 10^9$ via the modular recurrence (trailing digits).
\item For each $k$, first check if the trailing 9 digits are pandigital (cheap filter).
\item If so, compute the leading 9 digits via the logarithmic formula of Theorem~\ref{thm:leading}.
\item Return the first $k$ where both checks pass.
\end{enumerate}

\section{Complexity}

\begin{itemize}
\item \textbf{Time:} $O(k^*)$ where $k^* = 329{,}468$. Each step is $O(1)$.
\item \textbf{Space:} $O(1)$.
\end{itemize}

\section{Answer}

\[
\boxed{329468}
