\section{Problem Statement}

Given integer vectors $\mathbf{a}, \mathbf{b}, \mathbf{c} \in \mathbb{Z}^3$, the scalar triple product is $\mathbf{a} \cdot (\mathbf{b} \times \mathbf{c}) = \det[\mathbf{a}|\mathbf{b}|\mathbf{c}]$. Compute a sum involving triple products of lattice vectors satisfying certain conditions, modulo $10^9+7$.

\section{Mathematical Foundation}

\begin{definition}
For $\mathbf{a} = (a_1,a_2,a_3)$, $\mathbf{b} = (b_1,b_2,b_3)$, $\mathbf{c} = (c_1,c_2,c_3)$:
\[
\mathbf{a}\cdot(\mathbf{b}\times\mathbf{c}) = \det\begin{pmatrix} a_1 & b_1 & c_1 \\ a_2 & b_2 & c_2 \\ a_3 & b_3 & c_3 \end{pmatrix}.
\]
\end{definition}

\begin{theorem}[Properties of the Scalar Triple Product]
The scalar triple product satisfies:
\begin{enumerate}
\item $\mathbf{a}\cdot(\mathbf{b}\times\mathbf{c}) = \mathbf{b}\cdot(\mathbf{c}\times\mathbf{a}) = \mathbf{c}\cdot(\mathbf{a}\times\mathbf{b})$ (cyclic invariance).
\item $\mathbf{a}\cdot(\mathbf{b}\times\mathbf{c}) = -\mathbf{a}\cdot(\mathbf{c}\times\mathbf{b})$ (transposition negation).
\item $|\mathbf{a}\cdot(\mathbf{b}\times\mathbf{c})|$ equals the volume of the parallelepiped.
\item $\mathbf{a}\cdot(\mathbf{b}\times\mathbf{c})=0$ if and only if $\mathbf{a},\mathbf{b},\mathbf{c}$ are coplanar.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Cyclic permutations of columns multiply $\det$ by $(-1)^2=1$.
(2) A single transposition multiplies $\det$ by $-1$.
(3) The parallelepiped volume is $|\det M|$ where $M=[\mathbf{a}|\mathbf{b}|\mathbf{c}]$, a standard result from multilinear algebra.
(4) $\det M = 0$ iff the columns are linearly dependent, i.e., coplanar through the origin.
\end{proof}

\begin{theorem}[Sublattice Index]
Let $\Lambda \subseteq \mathbb{Z}^3$ be the sublattice generated by $\mathbf{a},\mathbf{b},\mathbf{c}\in\mathbb{Z}^3$. Then $[\mathbb{Z}^3:\Lambda]=|\det[\mathbf{a}|\mathbf{b}|\mathbf{c}]|$.
\end{theorem}

\begin{proof}
The Smith normal form of $M=[\mathbf{a}|\mathbf{b}|\mathbf{c}]$ gives $M=UDV$ with $U,V\in\mathrm{GL}_3(\mathbb{Z})$ and $D=\mathrm{diag}(d_1,d_2,d_3)$. The index $[\mathbb{Z}^3:\Lambda]=d_1 d_2 d_3 = |\det D|=|\det M|$, since $|\det U|=|\det V|=1$.
\end{proof}

\begin{lemma}[Primitive Vector Density]
The number of primitive vectors $\mathbf{v}\in\mathbb{Z}^3$ with $\|\mathbf{v}\|_\infty \le N$ satisfies
\[
P(N) = \frac{(2N+1)^3}{\zeta(3)} + O(N^2 \log N).
\]
\end{lemma}

\begin{proof}
By M\"obius inversion, $P(N) = \sum_{d=1}^{2N}\mu(d)\lfloor(2N+1)/d\rfloor^3$. The main term arises from $\sum_{d=1}^{\infty}\mu(d)/d^3 = 1/\zeta(3)$, with error term $O(N^2\log N)$ by standard analytic number theory.
\end{proof}

\section{Algorithm}

Enumerate lattice vector triples with bounded coordinates, compute determinants via the $3\times 3$ formula, apply problem-specific filters, and accumulate the sum modulo $10^9+7$. Exploit cyclic invariance and sign symmetries to reduce enumeration.

\item \textbf{Space:} $O(1)$ auxiliary.
\end{itemize}
