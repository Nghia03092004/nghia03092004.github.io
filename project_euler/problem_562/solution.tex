\section{Problem Statement}

Given an $N \times N$ grid of integer lattice points $\{0,1,\ldots,N\}^2$, find the triangle with vertices on lattice points having maximum perimeter. Define $f(N)$ as this maximum perimeter (rounded to $10^{-4}$). Find $f(50)$.

\section{Mathematical Foundation}

\begin{theorem}[Distance on Integer Lattice]
The Euclidean distance between lattice points $(x_1,y_1)$ and $(x_2,y_2)$ is $\sqrt{(x_2-x_1)^2+(y_2-y_1)^2}$. The perimeter of triangle $ABC$ is $P = |AB|+|BC|+|CA|$.
\end{theorem}

\begin{proof}
Immediate from the Euclidean metric.
\end{proof}

\begin{lemma}[Near-Diameter Edges Dominate]
In an optimal perimeter triangle on $\{0,\ldots,N\}^2$, all three vertices lie on the boundary of the grid.
\end{lemma}

\begin{proof}
If a vertex~$V$ lies in the interior at distance $\delta > 0$ from the boundary, moving~$V$ to a suitable boundary point increases at least one edge without reducing the others by more than~$\delta$. Iterating shows all vertices may be taken on the boundary.
\end{proof}

\begin{theorem}[Optimal Configuration]
The maximum-perimeter lattice triangle inscribed in $\{0,\ldots,N\}^2$ has vertices near three distinct sides of the square, approximating the continuous optimum (an equilateral-like triangle inscribed in the square).
\end{theorem}

\begin{proof}
Among continuous triangles inscribed in a square, the perimeter is maximized by a configuration with vertices on three sides. The lattice constraint restricts to $O(N)$ boundary points, and exhaustive search over $\binom{4N}{3}$ triples identifies the optimum.
\end{proof}

\section{Algorithm}

\begin{enumerate}
\item Enumerate all $4N$ boundary lattice points of $\{0,\ldots,N\}^2$.
\item For each triple of boundary points, compute the perimeter.
\item Track and return the maximum.
\end{enumerate}

Pruning: fix the longest edge candidate (near-diagonal pairs) and search only over third vertices that can improve the current best.

\section{Complexity Analysis}

\begin{itemize}
\item \textbf{Time:} $O(N^3)$ from $\binom{4N}{3}$ boundary triples. For $N=50$, approximately $1.3\times10^6$ triples.
\item \textbf{Space:} $O(N)$ for boundary point storage.
\end{itemize}
