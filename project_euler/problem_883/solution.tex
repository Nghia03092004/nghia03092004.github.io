\section{Problem Statement}
Study irreducible representations of finite groups and the dimension identity.

\section{Key Results}

\begin{theorem}[Dimension Formula]
$\sum_{i=1}^{h} (\dim \rho_i)^2 = |G|$.
\end{theorem}

\begin{proof}
The regular representation has dimension $|G|$ and decomposes as $\bigoplus (\dim\rho_i)\cdot\rho_i$.
\end{proof}

\begin{theorem}[Orthogonality]
$\frac{1}{|G|}\sum_{g} \chi_i(g)\overline{\chi_j(g)} = \delta_{ij}$.
\end{theorem}
