\section{Mathematical Framework}

\begin{theorem}[Gaussian Prime Classification]
$\pi \in \mathbb{Z}[i]$ is prime iff: (1) $N(\pi) = 2$, or (2) $N(\pi) = p \equiv 1 \pmod{4}$, or (3) $\pi = p$ with $p \equiv 3 \pmod{4}$.
\end{theorem}

\begin{theorem}[Fermat Two-Square]
A prime $p$ is $a^2+b^2$ iff $p = 2$ or $p \equiv 1 \pmod{4}$.
\end{theorem}
