\section{Problem Statement}
Compute $\pi_2(N) = |\{p \le N : p, p+2 \in \mathbb{P}\}|$.

\section{Theory}

\begin{theorem}[Brun, 1919]
$\displaystyle\sum_{\substack{p,p+2\in\mathbb{P}}} \left(\frac{1}{p}+\frac{1}{p+2}\right) < \infty.$
\end{theorem}

\begin{conjecture}[Hardy--Littlewood]
$\pi_2(N) \sim 2C_2 \frac{N}{(\ln N)^2}$ where $C_2 = \prod_{p\ge3}\frac{p(p-2)}{(p-1)^2} \approx 0.66016$.
\end{conjecture}
