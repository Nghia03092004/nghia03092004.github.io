\section{Problem Statement}
Compute $\sum_{m=2}^{N}\pi(m)$ where $\pi(m)$ is the Pisano period.
\begin{theorem}
$\pi(m) = \mathrm{lcm}(\pi(p_1^{a_1}), \ldots, \pi(p_k^{a_k}))$ and $\pi(p^k) = p^{k-1}\pi(p)$.
\end{theorem}
\begin{theorem}
$\pi(m) \le 6m$ for all $m$.
\end{theorem}
