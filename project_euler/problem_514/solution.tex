\section{Problem Statement}

Count the number of distinct shapes (simple closed lattice polygons) on an $n \times n$ geoboard, where two shapes are equivalent if one can be mapped to the other by a translation, rotation, or reflection.

\section{Mathematical Foundation}

\begin{theorem}[Burnside's Lemma]
Let a finite group $G$ act on a finite set $X$. The number of distinct orbits is
\[
|X/G| = \frac{1}{|G|}\sum_{g \in G}|X^g|,
\]
where $X^g = \{x \in X : g \cdot x = x\}$.
\end{theorem}

\begin{proof}
Count pairs $(g,x) \in G \times X$ with $g \cdot x = x$ in two ways. Over~$g$: $\sum_g |X^g|$. Over~$x$: $\sum_x |\mathrm{Stab}(x)|$. By the orbit-stabilizer theorem, $|\mathrm{Stab}(x)| = |G|/|\mathrm{Orb}(x)|$. Grouping by orbits:
\[
\sum_{x \in X}\frac{|G|}{|\mathrm{Orb}(x)|} = |G| \cdot |\{\text{orbits}\}|.
\]
Equating gives the result.
\end{proof}

\begin{theorem}[Symmetry Group]
The equivalence group on the finite grid is generated by the dihedral group $D_4$ (order~8) and lattice translations. Shapes equivalent under both are identified.
\end{theorem}

\begin{proof}
The symmetries of $\mathbb{Z}^2$ preserving the square lattice form the wallpaper group $p4m$. Restricted to a finite grid, the relevant symmetries are translations (within the grid) and the eight elements of $D_4$.
\end{proof}

\begin{lemma}[Canonical Position]
Fixing a canonical position (e.g., lexicographically smallest vertex at the origin) quotients out translations. Burnside's lemma over $D_4$ alone then counts the orbits correctly.
\end{lemma}

\begin{proof}
After canonicalization, two shapes are equivalent iff one is the image of the other under some $g \in D_4$ (followed by re-canonicalization). Thus Burnside over $D_4$ on the canonical set gives the orbit count.
\end{proof}

\section{Algorithm}

\begin{enumerate}
    \item Enumerate all simple lattice polygons on the $n \times n$ grid.
    \item Canonicalize each polygon by translation.
    \item Apply Burnside's lemma over $D_4$: for each $g \in D_4$, count the canonical polygons fixed by $g$ (after re-canonicalization).
    \item Return $\frac{1}{8}\sum_{g \in D_4}|X^g|$.
\end{enumerate}

\section{Complexity Analysis}

\begin{itemize}
    \item \textbf{Time:} Exponential in $n^2$. Feasible only for small~$n$.
    \item \textbf{Space:} $O(|X|)$, exponential in $n^2$.
\end{itemize}
