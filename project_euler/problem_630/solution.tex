\section{Problem Statement}
Generate points via $s_{n+1} = s_n^2 \bmod 50515093$, $s_0 = 290797$. Form segments and count proper crossings.

\section{Orientation Test}
\begin{theorem}
Segments $\overline{AB}$ and $\overline{CD}$ properly cross iff
\[
\mathrm{sign}(\mathrm{ccw}(A,C,D)) \ne \mathrm{sign}(\mathrm{ccw}(B,C,D)) \;\;\text{and}\;\; \mathrm{sign}(\mathrm{ccw}(A,B,C)) \ne \mathrm{sign}(\mathrm{ccw}(A,B,D))
\]
where $\mathrm{ccw}(P,Q,R) = (Q_x-P_x)(R_y-P_y) - (Q_y-P_y)(R_x-P_x)$.
\end{theorem}

\begin{proof}
Standard computational geometry: the condition is equivalent to $A,B$ being on opposite sides of line~$CD$ and $C,D$ being on opposite sides of line~$AB$.
\end{proof}

\section{Complexity}
\begin{itemize}
\item Brute force: $O(n^2)$ for $n$ segments.
\item Sweep line: $O((n+k)\log n)$.
\end{itemize}
