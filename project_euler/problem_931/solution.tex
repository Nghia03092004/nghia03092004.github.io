\section{Problem Statement}

For a positive integer $n$, define $R(n) = \sum_{d \mid n} \operatorname{rad}(d)$, where $\operatorname{rad}(d)$ is the product of the distinct prime factors of $d$ (with $\operatorname{rad}(1) = 1$). Find $\sum_{n=1}^{10^7} R(n) \bmod (10^9 + 7)$.

\section{Mathematical Foundation}

\begin{definition}
For $n \geq 1$, the \emph{radical} of $n$ is $\operatorname{rad}(n) = \prod_{p \mid n} p$, with $\operatorname{rad}(1) = 1$.
\end{definition}

\begin{lemma}
The radical function $\operatorname{rad}$ is multiplicative: if $\gcd(a, b) = 1$, then $\operatorname{rad}(ab) = \operatorname{rad}(a) \cdot \operatorname{rad}(b)$.
\end{lemma}

\begin{proof}
Let $a = \prod p_i^{a_i}$ and $b = \prod q_j^{b_j}$ where $\{p_i\}$ and $\{q_j\}$ are disjoint (since $\gcd(a,b) = 1$). Then the set of distinct prime factors of $ab$ is $\{p_i\} \cup \{q_j\}$. Hence $\operatorname{rad}(ab) = \prod p_i \cdot \prod q_j = \operatorname{rad}(a) \cdot \operatorname{rad}(b)$.
\end{proof}

\begin{theorem}
The function $R(n) = \sum_{d \mid n} \operatorname{rad}(d)$ is multiplicative, and for a prime power $p^a$ with $a \geq 1$:
\[
R(p^a) = 1 + ap.
\]
\end{theorem}

\begin{proof}
Since $R = \operatorname{rad} * \mathbf{1}$ is the Dirichlet convolution of two multiplicative functions, $R$ is multiplicative. The divisors of $p^a$ are $1, p, p^2, \ldots, p^a$. We have $\operatorname{rad}(1) = 1$ and $\operatorname{rad}(p^k) = p$ for all $k \geq 1$. Therefore:
\[
R(p^a) = \sum_{k=0}^{a} \operatorname{rad}(p^k) = 1 + \sum_{k=1}^{a} p = 1 + ap.
\]
\end{proof}

\begin{corollary}
For $n = p_1^{a_1} \cdots p_r^{a_r}$:
\[
R(n) = \prod_{i=1}^{r} (1 + a_i \, p_i).
\]
\end{corollary}

\begin{theorem}[Summation by Dirichlet interchange]
For any $N \geq 1$:
\[
\sum_{n=1}^{N} R(n) = \sum_{d=1}^{N} \operatorname{rad}(d) \cdot \left\lfloor \frac{N}{d} \right\rfloor.
\]
\end{theorem}

\begin{proof}
We have:
\[
\sum_{n=1}^{N} R(n) = \sum_{n=1}^{N} \sum_{d \mid n} \operatorname{rad}(d) = \sum_{d=1}^{N} \operatorname{rad}(d) \sum_{\substack{n=1 \\ d \mid n}}^{N} 1 = \sum_{d=1}^{N} \operatorname{rad}(d) \cdot \left\lfloor \frac{N}{d} \right\rfloor,
\]
where the second equality follows by swapping the order of summation.
\end{proof}

\section{Algorithm}

\begin{enumerate}
\item Compute $\operatorname{rad}(d)$ for all $d \leq N$ via a linear sieve ($O(N \log \log N)$).
\item Compute $\sum_{d=1}^{N} \operatorname{rad}(d) \cdot \lfloor N/d \rfloor \bmod (10^9+7)$.
\end{enumerate}

\section{Complexity Analysis}

\begin{itemize}
\item \textbf{Time:} $O(N \log \log N)$ for the sieve; $O(N)$ for the summation. Overall: $O(N \log \log N)$.
\item \textbf{Space:} $O(N)$ for the radical array.
\end{itemize}
