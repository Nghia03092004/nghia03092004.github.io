\section{Problem Statement}

Let $d_1 d_2 \cdots d_{10}$ be a $0$-to-$9$ pandigital number. Define the first seven primes $p_1 = 2, \ldots, p_7 = 17$. We require $p_i \mid \overline{d_{i+1}\,d_{i+2}\,d_{i+3}}$ for $i = 1, \ldots, 7$. Find the sum of all such pandigital numbers.

\section{Definitions}

\begin{definition}
For digits $d_i, d_j, d_k$, the \emph{three-digit substring value} is $\overline{d_i d_j d_k} = 100 d_i + 10 d_j + d_k$.
\end{definition}

\section{Structural Lemmas}

\begin{lemma}[Parity of $d_4$]
$d_4 \in \{0, 2, 4, 6, 8\}$.
\end{lemma}

\begin{proof}
$2 \mid \overline{d_2 d_3 d_4}$ requires $2 \mid d_4$, since $100 d_2 + 10 d_3$ is even.
\end{proof}

\begin{lemma}[Constraint on $d_6$]
$d_6 \in \{0, 5\}$.
\end{lemma}

\begin{proof}
$5 \mid \overline{d_4 d_5 d_6}$ requires $5 \mid d_6$, since $100 d_4 + 10 d_5 \equiv 0 \pmod{5}$.
\end{proof}

\begin{lemma}[Divisibility by 3]
$3 \mid \overline{d_3 d_4 d_5} \iff d_3 + d_4 + d_5 \equiv 0 \pmod{3}$.
\end{lemma}

\begin{proof}
Since $100 \equiv 10 \equiv 1 \pmod{3}$, we have $\overline{d_3 d_4 d_5} \equiv d_3 + d_4 + d_5 \pmod{3}$.
\end{proof}

\begin{lemma}[Overlapping constraint propagation]
Consecutive divisibility conditions share two digits, enabling efficient right-to-left construction: fixing $d_{i+2}, d_{i+3}$ reduces each extension step to a single digit choice.
\end{lemma}

\section{Main Result}

\begin{theorem}[Exhaustive enumeration]
There are exactly six $0$-to-$9$ pandigital numbers satisfying the substring divisibility property:
\begin{align*}
&1406357289, \quad 1430952867, \quad 1460357289, \\
&4106357289, \quad 4130952867, \quad 4160357289.
\end{align*}
\end{theorem}

\begin{proof}
We construct all solutions by right-to-left extension with constraint propagation:

\emph{Step 1.} Enumerate triples $(d_8, d_9, d_{10})$ with $17 \mid \overline{d_8 d_9 d_{10}}$ and pairwise distinct digits. There are $\lfloor 999/17 \rfloor = 58$ multiples of 17 in $[0,999]$; after distinct-digit filtering, at most 53 survive.

\emph{Steps 2--7.} For each surviving candidate, extend leftward by choosing an unused digit satisfying divisibility by $13, 11, 7, 5, 3, 2$ respectively. At each step, Lemmas~1--3 provide additional structural constraints that aggressively prune the search.

\emph{Step 8.} Assign the sole remaining digit to $d_1$.

This exhaustive procedure yields exactly six solutions. Their sum:
\[
\sum = 1406357289 + 1430952867 + 1460357289 + 4106357289 + 4130952867 + 4160357289 = 16695334890. \qedhere
\]
\end{proof}

\section{Algorithm}

\begin{algorithm}[H]
\caption{Substring-Divisible Pandigitals}
\begin{algorithmic}[1]
\State $\text{primes} \gets [17, 13, 11, 7, 5, 3, 2]$
\State Seed candidates: all $(d_8, d_9, d_{10})$ with $17 \mid \overline{d_8 d_9 d_{10}}$, distinct digits
\For{$p \in \{13, 11, 7, 5, 3, 2\}$}
    \For{each candidate $(\text{seq}, \text{used})$}
        \For{$d \in \{0,\ldots,9\} \setminus \text{used}$}
            \If{$p \mid (100d + 10 \cdot \text{seq}[0] + \text{seq}[1])$}
                \State extend candidate with $d$
            \EndIf
        \EndFor
    \EndFor
\EndFor
\State Assign remaining digit as $d_1$; sum all resulting numbers
\end{algorithmic}
\end{algorithm}

\section{Complexity Analysis}

\begin{proposition}[Time]
$O(C)$ where $C$ is the total number of partial candidates explored. Empirically $C = O(10^2)$ due to cascading constraints, versus $O(10!) \approx 3.6 \times 10^6$ for brute force.
\end{proposition}

\begin{proposition}[Space]
$O(C)$ for storing the candidate list.
\end{proposition}

\section{Answer}

\[
\boxed{16695334890}
\]
