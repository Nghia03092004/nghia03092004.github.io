\title{Problem 174: Counting the Number of ``Hollow'' Square Laminae That Can Form One, Two, Three, \ldots\ Distinct Arrangements}
\author{Project Euler Solution}

\section{Problem Statement}

Define $N(t)$ as the number of hollow square laminae that can be formed using exactly $t$ tiles. How many values of $t \leq 10^6$ satisfy $1 \leq N(t) \leq 10$?

\section{Mathematical Foundation}

\begin{theorem}[Tile-count parameterization]
Every hollow square lamina with outer side $a$, inner side $b$, $a \equiv b \pmod{2}$, $a \geq b + 2$, $b \geq 1$, uses $t = 4k(b + k)$ tiles where $k = (a - b)/2 \geq 1$. In particular, $4 \mid t$.
\end{theorem}

\begin{proof}
From Problem 173, $t = a^2 - b^2 = (a - b)(a + b)$. Setting $a - b = 2k$ gives $a + b = 2b + 2k$, so $t = 2k(2b + 2k) = 4k(b + k)$. Since $k \geq 1$ and $b + k \geq 2$, we have $t \geq 8$ and $4 \mid t$. \qed
\end{proof}

\begin{theorem}[Reduction to divisor counting]
Write $t = 4m$. Then $N(t)$ equals the number of divisors $k$ of $m$ satisfying $1 \leq k < \sqrt{m}$.
\end{theorem}

\begin{proof}
Setting $q = b + k$, the equation $m = k \cdot q$ must hold with $q > k$ (since $b = q - k \geq 1$ requires $q > k$). Each such factorization gives a unique lamina. The count is:
\[
N(4m) = \begin{cases} (d(m) - 1)/2 & \text{if } m \text{ is a perfect square} \\ d(m)/2 & \text{otherwise} \end{cases}
\]
which in both cases equals $|\{k : k \mid m,\; k < \sqrt{m}\}|$. \qed
\end{proof}

\begin{lemma}[Divisor sieve]
For $M = N/4$, the array $N[1 \ldots M]$ can be computed in $O(M \log M)$ time by iterating $k = 1, 2, \ldots, \lfloor\sqrt{M}\rfloor$ and for each $k$, incrementing $N[kq]$ for all $q = k + 1, k + 2, \ldots$ with $kq \leq M$.
\end{lemma}

\begin{proof}
Each pair $(k, q)$ with $k < q$ and $kq = m$ is visited exactly once. The total number of pairs is $\sum_{k=1}^{\lfloor\sqrt{M}\rfloor} \lfloor M/k - k \rfloor = O(M \log M)$. \qed
\end{proof}

\section{Algorithm}

\begin{algorithmic}[1]
\State $M \gets N/4 = 250000$
\State Initialize $\text{count}[1 \ldots M] \gets 0$
\For{$k = 1$ to $\lfloor\sqrt{M}\rfloor$}
    \For{$q = k+1$ to $\lfloor M/k \rfloor$}
        \State $\text{count}[k \cdot q] \gets \text{count}[k \cdot q] + 1$
    \EndFor
\EndFor
\State $\text{answer} \gets |\{m : 1 \leq \text{count}[m] \leq 10\}|$
\end{algorithmic}

\section{Complexity Analysis}

\begin{itemize}
    \item \textbf{Time:} $O(M \log M)$ where $M = 250{,}000$.
    \item \textbf{Space:} $O(M)$ for the count array.
\end{itemize}

\section{Answer}

\[
\boxed{209566}
