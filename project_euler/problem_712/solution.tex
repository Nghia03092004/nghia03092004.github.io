\section{Problem Statement}

For integers $n \ge 2$ and $m \ge 1$, let $v_n(k)$ denote the $n$-adic valuation of~$k$. Define
\[
D(n,m) = \sum_{1 \le i < j \le m} |v_n(i) - v_n(j)|.
\]
Compute $\displaystyle\sum_{\substack{p \le N \\ p \text{ prime}}} D(p,p) \pmod{10^9+7}$ for the given~$N$.

\section{Mathematical Foundation}

\begin{lemma}
For a prime~$p$ and integer~$m$, the number of integers in $\{1,\ldots,m\}$ with $p$-adic valuation exactly~$e$ is
\[
c_e = \left\lfloor \frac{m}{p^e} \right\rfloor - \left\lfloor \frac{m}{p^{e+1}} \right\rfloor.
\]
\end{lemma}

\begin{proof}
The count of $k \le m$ with $p^e \mid k$ is $\lfloor m/p^e \rfloor$. Subtracting those with $v_p(k) \ge e+1$ gives the result.
\end{proof}

\begin{theorem}
$D(p,m) = \displaystyle\sum_{0 \le e_1 < e_2} (e_2 - e_1)\, c_{e_1}\, c_{e_2}$.
\end{theorem}

\begin{proof}
Group pairs $(i,j)$ by their valuations $(e_1, e_2)$ with $e_1 < e_2$. Each such group contributes $(e_2 - e_1) \cdot c_{e_1} \cdot c_{e_2}$ to the sum.
\end{proof}

\begin{lemma}
For $m = p$, we have $c_0 = p-1$, $c_1 = 1$, and $c_e = 0$ for $e \ge 2$. Hence $D(p,p) = p-1$.
\end{lemma}

\begin{proof}
Among $\{1,\ldots,p\}$, only $k = p$ has $v_p(k) = 1$; all others have $v_p(k) = 0$. Since $p^2 > p$, no element has valuation~$\ge 2$. Thus $D(p,p) = (1-0)(p-1)(1) = p-1$.
\end{proof}

\begin{theorem}
The answer is $\displaystyle\sum_{\substack{p \le N \\ p \text{ prime}}} (p-1) = \biggl(\sum_{\substack{p \le N \\ p \text{ prime}}} p\biggr) - \pi(N)$.
\end{theorem}

\begin{proof}
Direct substitution and linearity of summation.
\end{proof}

\section{Algorithm}

\begin{verbatim}
function solve(N, mod):
    is_prime = sieve_of_eratosthenes(N)
    result = 0
    for p = 2 to N:
        if is_prime[p]:
            result = (result + p - 1) mod mod
    return result
\end{verbatim}

\section{Complexity Analysis}

\begin{itemize}
    \item \textbf{Time:} $O(N \log\log N)$ for the sieve, $O(N)$ for summation.
    \item \textbf{Space:} $O(N)$ for the sieve array.
\end{itemize}
