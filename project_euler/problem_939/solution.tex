\section{Problem Statement}

For an even number $n \geq 4$, let $G(n)$ be the number of ways to write $n = p + q$ where $p \leq q$ are both prime. Find $\sum_{\substack{n=4 \\ n \text{ even}}}^{10^6} G(n)$.

\section{Mathematical Foundation}

\begin{definition}
For even $n \geq 4$, the \emph{Goldbach partition function} is:
\[
G(n) = \#\{(p, q) : p + q = n,\; p \leq q,\; p, q \text{ prime}\}.
\]
\end{definition}

\begin{theorem}[Goldbach's conjecture -- computational verification]
Every even integer $n$ with $4 \leq n \leq 4 \times 10^{18}$ satisfies $G(n) \geq 1$.
\end{theorem}

\begin{proof}
Verified computationally by Oliveira e Silva (2013) up to $4 \times 10^{18}$.
\end{proof}

\begin{theorem}[Sieve-based computation]
\[
G(n) = \#\{p \in \mathcal{P} : p \leq n/2,\; n - p \in \mathcal{P}\}.
\]
\end{theorem}

\begin{proof}
Each unordered pair $\{p, q\}$ with $p + q = n$ and $p \leq q$ corresponds uniquely to a prime $p \leq n/2$ with $q = n - p$ also prime.
\end{proof}

\begin{lemma}[Counting identity]
$\sum_{\substack{n=4 \\ n \text{ even}}}^{N} G(n)$ counts all unordered pairs of primes $(p, q)$ with $p \leq q$ and $p + q \leq N$ even.
\end{lemma}

\begin{proof}
Each pair $(p, q)$ with $p \leq q$ both prime and $p + q$ even is counted exactly once in $G(p + q)$. Summing over even $n$ from 4 to $N$ collects all such pairs.
\end{proof}

\begin{theorem}[Hardy--Littlewood conjecture]
For even $n \to \infty$:
\[
G(n) \sim \frac{2 C_2 \, n}{(\ln n)^2} \prod_{\substack{p \mid n \\ p > 2}} \frac{p - 1}{p - 2},
\]
where $C_2 = \prod_{p > 2}\left(1 - \frac{1}{(p-1)^2}\right) \approx 0.6602$.
\end{theorem}

\begin{proof}
Heuristic conjecture from the Hardy--Littlewood circle method; a rigorous proof remains open.
\end{proof}

\section{Algorithm}

\begin{enumerate}
\item Sieve primes up to $N$ via Eratosthenes.
\item For each even $n$ from 4 to $N$: count primes $p \leq n/2$ with $n - p$ also prime.
\item Accumulate $G(n)$ into the total.
\end{enumerate}

\section{Complexity Analysis}

\begin{itemize}
\item \textbf{Time:} $O(N \log \log N)$ for the sieve; $O(N^2)$ for the counting (or $O(N^2 / \ln^2 N)$ iterating over prime pairs).
\item \textbf{Space:} $O(N)$ for the primality array.
\end{itemize}
