\section{Problem Statement}

For every real number $a > 1$, define:
\begin{align*}
g_a(x) &= 1 \quad \text{for } x < a, \\
g_a(x) &= g_a(x-1) + g_a(x-a) \quad \text{for } x \geq a.
\end{align*}
Let $G(n) = g_{\sqrt{n}}(n)$. Given $G(90) = 7564511$, find $\sum G(p) \bmod (10^9+7)$ for all primes $p$ with $10^7 < p < 10^7 + 10^4$.

\section{Mathematical Foundation}

\begin{theorem}[Binomial Coefficient Formula]
Let $a = \sqrt{p}$ and $K = \lfloor a \rfloor$. Then
\[
G(p) = \sum_{k=0}^{K}\binom{n_k}{k},
\]
where $n_k = p - 1 - \lfloor k(\sqrt{p} - 1)\rfloor$.
\end{theorem}

\begin{proof}
The value $g_a(p)$ counts the number of ways to reach $p$ from some starting value $x_0 < a$ using steps of size~1 and size~$a$. Taking $k$ steps of size $a$ consumes $ka$ total, leaving $p - ka$ to cover with unit steps, but the walk must also start from a value less than $a$. Unfolding the recursion, the number of interleavings of $k$ large steps among the total steps gives the binomial coefficient $\binom{n_k}{k}$, where $n_k$ accounts for the effective number of positions. Since $a$ is irrational, the exact floor accounting yields $n_k = p - 1 - \lfloor k(a - 1)\rfloor$.
\end{proof}

\begin{lemma}[Modular Binomial Computation]
For a prime modulus $q$, after $O(n)$ preprocessing of factorials and their modular inverses,
\[
\binom{n}{k} \equiv n!\,(k!)^{-1}\,((n-k)!)^{-1} \pmod{q}
\]
can be evaluated in $O(1)$ time per query.
\end{lemma}

\begin{proof}
Since $\mathbb{Z}/q\mathbb{Z}$ is a field, modular inverses exist for all nonzero elements. Precompute $i! \bmod q$ iteratively, then $(N!)^{-1}$ via Fermat's little theorem, and propagate downward: $(i!)^{-1} = ((i+1)!)^{-1} \cdot (i+1)$.
\end{proof}

\begin{theorem}[Segmented Sieve]
Primes in $(10^7, 10^7 + 10^4)$ can be found in $O(\sqrt{10^7} + 10^4)$ time via a segmented sieve.
\end{theorem}

\begin{proof}
Sieve small primes up to $\sqrt{10^7 + 10^4}$. For each, mark multiples in the target interval. Unmarked entries are prime.
\end{proof}

\section{Algorithm}

\begin{enumerate}
    \item Sieve primes in $(10^7, 10^7 + 10^4)$.
    \item Precompute factorials and inverse factorials modulo $10^9 + 7$ up to $\sim 10^7$.
    \item For each prime $p$: compute $G(p) = \sum_{k=0}^{\lfloor\sqrt{p}\rfloor}\binom{n_k}{k}$ using the precomputed tables.
    \item Sum all $G(p)$ modulo $10^9 + 7$.
\end{enumerate}

\section{Complexity Analysis}

\begin{itemize}
    \item \textbf{Time:} $O(10^7)$ for factorial precomputation; $O(620 \times 3162) \approx O(2 \times 10^6)$ for the main computation.
    \item \textbf{Space:} $O(10^7)$ for factorial arrays.
\end{itemize}
