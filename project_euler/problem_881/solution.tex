\section{Problem Statement}
Define $G_n$ on $\{1,\ldots,n\}$ with edges $i \sim j$ when $i \mid j$ or $j \mid i$. Find $\chi(G_n)$.

\section{Key Results}

\begin{theorem}
$\chi(G_n) = \lfloor \log_2 n \rfloor + 1$.
\end{theorem}

\begin{proof}
The chain $1, 2, 4, \ldots, 2^{\lfloor\log_2 n\rfloor}$ is a clique of size $\lfloor\log_2 n\rfloor + 1$, so $\chi \ge \lfloor\log_2 n\rfloor + 1$.

For the upper bound, assign $c(k) = \lfloor\log_2 k\rfloor$. If $i \mid j$ with $i < j$, then $j \ge 2i$, so $\lfloor\log_2 j\rfloor \ge \lfloor\log_2 i\rfloor + 1$, giving distinct colors.
\end{proof}

\begin{theorem}[Dilworth]
In a finite poset, max chain length = min antichain partition size.
\end{theorem}
