\section{Problem Statement}

Define $f : \mathbb{N}_0 \times \mathbb{N}_0 \to \mathbb{N}_0$ by
\[
f(n, k) = \begin{cases}
n & \text{if } k = 0, \\
f(n \oplus k,\, \lfloor k/2 \rfloor) & \text{if } k > 0,
\end{cases}
\]
where $\oplus$ is bitwise XOR. Let $S(N) = \sum_{k=0}^{N}\sum_{n=0}^{N} f(n,k)$. Compute $S(10^{18}) \bmod (10^9+7)$.

\section{Mathematical Foundation}

Let $B = \lceil \log_2(N+1) \rceil$ and write $k = \sum_{i=0}^{L} b_i 2^i$ in binary.

\begin{lemma}[Closed form for $f$]
\label{lem:closed}
For all $n, k \in \mathbb{N}_0$,
\[
f(n, k) = n \oplus g(k), \qquad g(k) := \bigoplus_{t=0}^{\infty} \lfloor k/2^t \rfloor.
\]
\end{lemma}

\begin{proof}
Define $k_t = \lfloor k/2^t \rfloor$. Unrolling the recursion, at step~$t$ the accumulator is $n \oplus k_0 \oplus k_1 \oplus \cdots \oplus k_{t-1}$ and the remaining key is~$k_t$. The recursion terminates when $k_t = 0$, yielding $f(n,k) = n \oplus \bigoplus_{t \ge 0} k_t = n \oplus g(k)$.
\end{proof}

\begin{lemma}[Suffix XOR characterisation]
\label{lem:suffix}
Bit~$j$ of $g(k)$ equals the suffix XOR parity:
\[
g(k)_j = \bigoplus_{i \ge j} b_i.
\]
\end{lemma}

\begin{proof}
Bit~$j$ of $\lfloor k/2^t \rfloor$ is $b_{j+t}$. Therefore $g(k)_j = \bigoplus_{t=0}^{L} b_{j+t} = \bigoplus_{i \ge j} b_i$.
\end{proof}

\begin{theorem}[Involution]
\label{thm:involution}
$g(g(k)) = k$ for all $k \in \mathbb{N}_0$.
\end{theorem}

\begin{proof}
Over $\mathrm{GF}(2)$, represent the binary digits of~$k$ as a column vector $\mathbf{b}$. The map~$g$ acts as $\mathbf{b} \mapsto M\mathbf{b}$ where $M$ is the upper-triangular all-ones matrix ($M_{ij} = 1$ for $j \ge i$). The $(i,j)$-entry of $M^2$ over $\mathrm{GF}(2)$ is $\bigoplus_{\ell=i}^{j} 1 = (j - i + 1) \bmod 2$, which equals~$1$ iff $j = i$. Hence $M^2 = I$ and $g$ is an involution.
\end{proof}

\begin{theorem}[Decomposition of $S(N)$]
\label{thm:decomp}
\[
S(N) = \sum_{j=0}^{B-1} 2^j \cdot C_j, \qquad C_j = A_j(N+1-B_j) + (N+1-A_j)B_j,
\]
where $A_j = |\{n \in [0,N] : n_j = 1\}|$ and $B_j = |\{k \in [0,N] : g(k)_j = 1\}|$.
\end{theorem}

\begin{proof}
Since $f(n,k) = n \oplus g(k)$, the sum decomposes by bit position:
\[
S(N) = \sum_{j} 2^j \cdot |\{(n,k) \in [0,N]^2 : (n \oplus g(k))_j = 1\}|.
\]
Bit~$j$ of $n \oplus g(k)$ is~1 iff $n_j \ne g(k)_j$. Since $n$ and $k$ are chosen independently, the count factors as $C_j = A_j(N+1-B_j) + (N+1-A_j)B_j$.
\end{proof}

\begin{lemma}[Bit counting]
\[
A_j = \left\lfloor \frac{N+1}{2^{j+1}} \right\rfloor \cdot 2^j + \max\!\bigl(0,\; (N+1) \bmod 2^{j+1} - 2^j\bigr).
\]
\end{lemma}

\begin{proof}
Among $\{0,\ldots,N\}$, bit~$j$ cycles with period $2^{j+1}$: off for $2^j$ values, on for $2^j$ values. Counting complete cycles plus the partial remainder yields the formula.
\end{proof}

\begin{lemma}[Digit DP for $B_j$]
The count $B_j$ is computed by a digit DP processing bits of $N$ from MSB to LSB, tracking a tight/free flag and the running suffix XOR parity from bit~$j$ onward. This requires $O(B)$ states per bit position~$j$.
\end{lemma}

\begin{proof}
Standard digit DP: at each bit level the state space is $\{0,1\} \times \{0,1\}$ (tight flag $\times$ parity), giving $O(1)$ states per level and $O(B)$ per target bit~$j$. Across all $j$, total work is $O(B^2)$.
\end{proof}

\section{Algorithm}

\begin{verbatim}
function solve(N):
    B = number of bits in N
    result = 0
    for j = 0 to B-1:
        A_j = bit_count(N, j)
        B_j = digit_dp_suffix_xor(N, j, B)
        C_j = A_j * (N+1 - B_j) + (N+1 - A_j) * B_j
        result += 2^j * C_j  (mod 10^9+7)
    return result
\end{verbatim}

\section{Complexity Analysis}

\begin{itemize}
\item \textbf{Time:} $O(B^2)$ where $B = \lceil \log_2(N+1) \rceil \approx 60$. Total $\approx 3600$ operations.
\item \textbf{Space:} $O(B)$ for DP state arrays.
\end{itemize}
