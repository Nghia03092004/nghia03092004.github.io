\section{Problem Statement}

For an $H \times W$ grid, let $S(G)$ be a graph-theoretic quantity over subgraphs~$G$. Define
\[
C(H,W) = \sum_{G} S(G).
\]
Given $C(3,3)=408$, find $C(10000,20000) \bmod (10^9+7)$.

\section{Mathematical Foundation}

\begin{definition}
A \emph{column state} is a vector $\mathbf{s} \in \{0,1\}^H$ encoding vertical edges in a column. The \emph{transfer matrix} $M$ encodes valid transitions between adjacent columns.
\end{definition}

\begin{theorem}[Transfer Matrix Method]
\[
C(H,W) = \mathbf{v}_0^\top \cdot M^{W-1} \cdot \mathbf{v}_f,
\]
where $M$ is $2^H \times 2^H$, $\mathbf{v}_0$ the initial vector, $\mathbf{v}_f$ the final vector.
\end{theorem}

\begin{proof}
Process the grid column by column. Each column transition depends only on adjacent column states. There are $W-1$ transitions, yielding $M^{W-1}$.
\end{proof}

\begin{lemma}
Direct transfer matrix is infeasible for $H = 10000$ since the state space has $2^H$ elements.
\end{lemma}

\begin{proof}
Each column has $H$ vertical edge positions, giving $2^H$ states.
\end{proof}

\begin{theorem}[State Reduction]
If $S(G)$ depends only on connectivity, states reduce to partitions of boundary nodes. The state count becomes the Bell number~$B_H$. For $H = 10000$, a closed-form expression in $H$ and~$W$ is required.
\end{theorem}

\begin{proof}
Column states inducing identical boundary-node partitions yield identical future contributions. By a Myhill--Nerode-type argument, the transfer matrix quotients to $B_H$ states. For large~$H$, algebraic simplification of the characteristic polynomial or a direct closed form is needed.
\end{proof}

\begin{lemma}
If $C(H,W) = P(H,W)$ for a closed-form~$P$, evaluation takes $O(\mathrm{poly}(\log H, \log W))$ time.
\end{lemma}

\begin{proof}
Modular arithmetic and fast exponentiation suffice.
\end{proof}

\section{Algorithm}

\begin{verbatim}
function C(H, W, mod):
    // Use closed-form or H-recurrence
    return closed_form(H, W, mod)

    // Alternative for small H:
    states = enumerate_partitions(H)
    M = build_transfer_matrix(states)
    return v0^T * M^(W-1) * vf mod mod
\end{verbatim}

\section{Complexity Analysis}

\begin{itemize}
    \item \textbf{Time:} $O(\mathrm{poly}(\log H, \log W))$ with closed form; $O(s^3 \log W)$ with transfer matrix of $s = B_H$ states.
    \item \textbf{Space:} $O(1)$ for closed form; $O(s^2)$ for transfer matrix.
\end{itemize}
