\section{Problem Statement}
Four points $A(a,0)$, $B(b,0)$, $C(0,c)$, $D(0,d)$ with $0 < a < b$, $0 < c < d$, integer coordinates. Point $P$ (integer coordinates) on line $AC$ such that triangles $ABP$, $CDP$, $BDP$ are all similar. This forces $a = c$.

Find the number of distinct triplets $(a,b,d)$ with $b + d < 10^8$ admitting a valid $P$.

Reference: $b+d<100 \Rightarrow 92$; $b+d<10^5 \Rightarrow 320\,471$.

\section{Mathematical Foundation}

\begin{theorem}[Constraint $a = c$]
For triangles $ABP$, $CDP$, $BDP$ to be simultaneously similar with $P$ on line $AC$, it is necessary that $a = c$.
\end{theorem}

\begin{proof}
The similarity conditions on three triangles sharing the point $P$ on line $AC$ impose constraints that, via coordinate geometry, force the right triangle $OAC$ (with $O$ the origin) to be isosceles, i.e., $a = c$.
\end{proof}

\begin{theorem}[Two Diophantine Cases]
With $a = c$ and $P = (a-t, t)$ on line $x + y = a$, the similarity conditions reduce to:
\begin{enumerate}
\item \textbf{Pythagorean case:} $b^2 + d^2 = e^2$ (perfect square hypotenuse), with $a$ determined by the incircle of triangle $OBD$.
\item \textbf{Eisenstein case:} $b^2 + bd + d^2 = f^2$ (Loeschian number), with $a$ determined by an alternate angle assignment.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{Case 1:} With $P = (r, r)$ as the incenter of $\triangle OBD$, the inradius is $r = bd/(b+d+\sqrt{b^2+d^2})$, requiring $b^2 + d^2$ to be a perfect square for $r$ (and hence $a = 2r$) to be a positive integer.

\emph{Case 2:} An alternate angle matching in the similarity yields $b^2 + bd + d^2 = f^2$. This is the norm form in $\mathbb{Z}[\omega]$ where $\omega = e^{2\pi i/3}$.
\end{proof}

\begin{lemma}[Pythagorean Triple Parametrization]
Primitive solutions to $b^2 + d^2 = e^2$: $b = m^2 - n^2$, $d = 2mn$, $e = m^2 + n^2$ for coprime $m > n > 0$ of opposite parity.
\end{lemma}

\begin{proof}
Classical; see Hardy and Wright, Chapter XX.
\end{proof}

\begin{lemma}[Loeschian Number Parametrization]
Primitive solutions to $b^2 + bd + d^2 = f^2$: $b = m^2 - n^2$, $d = 2mn + n^2$, $f = m^2 + mn + n^2$ for coprime $m > n > 0$ with $m \not\equiv n \pmod{3}$.
\end{lemma}

\begin{proof}
Follows from unique factorization in $\mathbb{Z}[\omega]$, where $N(a + b\omega) = a^2 + ab + b^2$.
\end{proof}

\section{Algorithm}

\begin{verbatim}
1. Case 1: Generate primitive Pythagorean triples (m,n) with
   m > n > 0, gcd(m,n) = 1, m+n odd. For each multiple k:
   compute b, d, e, then a from incircle formula.
2. Case 2: Generate primitive Eisenstein triples. For each
   multiple k: compute b, d, f, then a from geometry.
3. Collect distinct (a, b, d) from both cases, subtract overlaps.
\end{verbatim}

\section{Complexity Analysis}
\begin{itemize}
\item \textbf{Time:} $O(L)$ total for generating all primitive triples and multiples with $b + d < L = 10^8$.
\item \textbf{Space:} $O(|\text{result}|)$ for storing distinct triplets.
\end{itemize}
