\section{Problem Statement}
Let $d_n(x)$ be the $n$-th decimal digit of the fractional part of $x$. Define
\[
S(n) = \sum_{k=1}^{n} d_n(1/k).
\]
Given $S(7) = 10$ and $S(100) = 418$, find $S(10^7)$.

\section{Mathematical Analysis}

\subsection{Long Division Characterization}
The decimal expansion $1/k = 0.d_1 d_2 d_3 \ldots$ is generated by long division with remainder sequence:
\begin{align}
r_0 &= 1 \\
d_i &= \lfloor 10\, r_{i-1} / k \rfloor \\
r_i &= 10\, r_{i-1} \bmod k
\end{align}

\begin{theorem}
$r_i = 10^i \bmod k$ for all $i \geq 0$.
\end{theorem}
\begin{proof}
By induction: $r_0 = 1 = 10^0 \bmod k$. If $r_{i-1} = 10^{i-1} \bmod k$, then $r_i = 10 \cdot r_{i-1} \bmod k = 10^i \bmod k$.
\end{proof}

Therefore:
\[
d_n(1/k) = \left\lfloor \frac{10 \cdot (10^{n-1} \bmod k)}{k} \right\rfloor
\]

\subsection{Efficient Computation}
For $n = N = 10^7$, we compute $10^{N-1} \bmod k$ via binary exponentiation in $O(\log N)$ time per value of $k$.

\section{Complexity}
\begin{itemize}
    \item \textbf{Time:} $O(N \log N)$
    \item \textbf{Space:} $O(1)$
\end{itemize}

\section{Answer}

\[
\boxed{44967734}
\]
