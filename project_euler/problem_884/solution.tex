\section{Problem Statement}
Count restricted lattice paths using the reflection principle and LGV lemma.

\section{Key Results}

\begin{theorem}[Reflection Principle]
Paths $(0,0) \to (n,n)$ staying below $y=x$: $C_n = \frac{1}{n+1}\binom{2n}{n}$.
\end{theorem}

\begin{theorem}[LGV Lemma]
Non-intersecting $k$-tuples of paths from sources $A$ to destinations $B$:
\[
\det\left[\binom{(b_j - a_i)_x + (b_j - a_i)_y}{(b_j - a_i)_x}\right]_{i,j=1}^{k}
\]
\end{theorem}

\begin{proof}
Sign-reversing involution cancels intersecting path systems in the determinant expansion.
\end{proof}
