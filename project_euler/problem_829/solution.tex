\section{Problem Statement}

Floor sum computation. See the accompanying solution.md for the full problem statement and mathematical analysis.

\section{Mathematical Framework}

\begin{theorem}
$R(n) = \sum_{k=1}^{n} (n \bmod k) = n^2 - \sum_{k=1}^{n} k\lfloor n/k \rfloor$.
\end{theorem}

\begin{theorem}[Floor Grouping]
$\lfloor n/k \rfloor$ takes $O(\sqrt{n})$ distinct values. Grouping by these values computes the sum in $O(\sqrt{n})$.
\end{theorem}

\begin{lemma}
$\sum_{k=1}^{n} k\lfloor n/k\rfloor = \sum_{m=1}^{n} \sigma_1(m)$ where $\sigma_1$ is the sum-of-divisors function.
\end{lemma}
