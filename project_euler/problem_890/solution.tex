\section{Problem Statement}
Compute the collision probability for $n$ items in $N$ bins.

\section{Key Results}

\begin{theorem}[Exact Formula]
$P(n,N) = 1 - \prod_{k=0}^{n-1}(1-k/N) = 1 - \frac{N!}{N^n(N-n)!}$.
\end{theorem}

\begin{theorem}[Poisson Approximation]
$P(n,N) \approx 1 - e^{-n(n-1)/(2N)}$.
\end{theorem}

\begin{theorem}[Threshold]
$P \ge 1/2$ when $n \ge \sqrt{2N\ln 2} \approx 1.177\sqrt{N}$.
\end{theorem}

\section{Example}
$P(23, 365) = 0.50730$: 23 people suffice for 50\% birthday collision.
