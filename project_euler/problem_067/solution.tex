\section{Problem Statement}

Given a triangle with 100 rows of integers, find the maximum total from top to bottom, where at each step you may move to an adjacent number on the row below.

\section{Analysis}

\subsection{Brute Force Infeasibility}

The number of distinct top-to-bottom paths in a triangle with $n$ rows is $2^{n-1}$. For $n = 100$:
\[
2^{99} \approx 6.34 \times 10^{29}
\]
This makes exhaustive search completely infeasible.

\subsection{Dynamic Programming Solution}

Let $T[i][j]$ denote the value at row $i$, column $j$ (both 0-indexed). Define $M[i][j]$ as the maximum path sum from cell $(i,j)$ to any cell in the last row.

\textbf{Base case:}
\[
M[n-1][j] = T[n-1][j], \quad 0 \le j \le n-1
\]

\textbf{Recurrence:}
\[
M[i][j] = T[i][j] + \max\bigl(M[i+1][j],\; M[i+1][j+1]\bigr), \quad 0 \le i \le n-2
\]

The answer is $M[0][0]$.

\begin{theorem}
The above recurrence correctly computes the maximum path sum.
\end{theorem}

\begin{proof}
By induction on $i$ from $n-1$ down to $0$.

\textbf{Base case} ($i = n-1$): Trivially, $M[n-1][j] = T[n-1][j]$ since there is no further row to descend to.

\textbf{Inductive step}: Suppose $M[k][j]$ is correct for all $k > i$ and all valid $j$. From cell $(i, j)$, the path must proceed to either $(i+1, j)$ or $(i+1, j+1)$. By the inductive hypothesis, the maximum achievable sum from those cells is $M[i+1][j]$ and $M[i+1][j+1]$, respectively. Thus:
\[
M[i][j] = T[i][j] + \max(M[i+1][j], M[i+1][j+1])
\]
is the maximum path sum starting from $(i, j)$.
\end{proof}

\section{Complexity}

\begin{itemize}
    \item \textbf{Time}: $O(n^2)$ where $n = 100$. Specifically, $\sum_{i=1}^{n-1} i = \frac{n(n-1)}{2} = 4950$ operations.
    \item \textbf{Space}: $O(1)$ extra if we modify the triangle in place; $O(n^2)$ for the triangle itself.
\end{itemize}

\section{Result}

The answer is $\boxed{7273}$.
