\section{Problem Statement}

Define:
\[
g(k) = 1 \quad \text{for } 0 \le k \le 1999
\]
\[
g(k) = g(k-2000) + g(k-1999) \quad \text{for } k \ge 2000
\]

Find $g(10^{18}) \bmod 20092010$.

\section{Characteristic Polynomial}

The recurrence $g(k) = g(k-2000) + g(k-1999)$ has characteristic polynomial:
\[
p(x) = x^{2000} - x - 1
\]

This follows from substituting $g(k) = x^k$ into the recurrence:
\[
x^k = x^{k-2000} + x^{k-1999} \implies x^{2000} = 1 + x \implies x^{2000} - x - 1 = 0
\]

\section{Polynomial Exponentiation}

\subsection{Method}

By Cayley--Hamilton, computing $g(10^{18})$ is equivalent to computing $x^{10^{18}} \bmod p(x)$ in the polynomial ring $\mathbb{Z}_m[x]/(p(x))$ where $m = 20092010$.

\subsection{Algorithm}

\begin{enumerate}
    \item Initialize $r(x) = 1$, $b(x) = x$.
    \item Binary exponentiation with $n = 10^{18}$: for each bit, square $b$ and conditionally multiply into $r$.
    \item Each multiplication: polynomial multiply (degree $\le 1999$), then reduce modulo $p(x)$.
    \item Reduction uses $x^{2000} \equiv x + 1$: for each term $c \cdot x^d$ with $d \ge 2000$:
    \[
    c \cdot x^d \equiv c \cdot x^{d-1999} + c \cdot x^{d-2000} \pmod{p(x)}
    \]
\end{enumerate}

\subsection{Extracting the Result}

Let $x^{10^{18}} \equiv \sum_{i=0}^{1999} a_i x^i \pmod{p(x)}$. Then:
\[
g(10^{18}) = \sum_{i=0}^{1999} a_i \cdot g(i) = \sum_{i=0}^{1999} a_i \pmod{20092010}
\]

since all initial values are 1.

\section{Complexity}

\begin{itemize}
    \item Polynomial multiplication: $O(n^2)$ per step, $n = 2000$.
    \item Reduction: $O(n)$ per step (sparse: only 2 additions per excess degree).
    \item Binary exponentiation: $O(\log(10^{18})) \approx 60$ steps.
    \item Total: $O(n^2 \log k) \approx 2.4 \times 10^8$ operations.
\end{itemize}

\section{Result}

\[
\boxed{g(10^{18}) \bmod 20092010 = 12747994}
\]

\section{Answer}

\[
\boxed{12747994}
\]
