\section{Problem Statement}
$\sum_{p \leq 10^6} (\text{sum of prim roots mod } p) \bmod 10^9+7$.

\section{Sum Formula}
\begin{theorem}
Sum of primitive roots mod $p$ $\equiv \mu(p-1) \pmod{p}$.
\end{theorem}

\begin{proof}
Primitive roots are $g^k$ with $\gcd(k,p-1)=1$. By Ramanujan sum theory, $\sum g^k = \mu(p-1) \pmod{p}$.
\end{proof}

\section{M\"obius Values}
$\mu(p-1) \in \{-1,0,1\}$: zero if $4|p-1$ (unless $p-1$ is squarefree), etc.

\section{Algorithm}
SPF sieve to factor $p-1$, compute $\mu(p-1)$, accumulate.
