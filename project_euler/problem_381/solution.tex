\section{Problem Statement}

For a prime $p$, let $S(p) = \sum_{k=1}^{5} (p-k)! \bmod p$. Find $\sum S(p)$ for all primes $5 \le p < 10^8$.

\section{Mathematical Foundation}

\begin{theorem}[Wilson's Theorem]
For any prime $p$,
\[
(p-1)! \equiv -1 \pmod{p}.
\]
\end{theorem}

\begin{proof}
In $\mathbb{Z}/p\mathbb{Z}$, every nonzero element has a unique multiplicative inverse. The only self-inverse elements are $\pm 1$ (roots of $x^2 \equiv 1$). In the product $(p-1)! = 1 \cdot 2 \cdots (p-1)$, all other elements cancel in inverse pairs, leaving $(p-1)! \equiv 1 \cdot (-1) = -1 \pmod{p}$.
\end{proof}

\begin{lemma}[Backward Factorial Recurrence]
For a prime $p \ge 5$ and $1 \le k \le 5$, the residues $(p-k)! \bmod p$ are:
\begin{center}
\begin{tabular}{cl}
\toprule
$k$ & $(p-k)! \bmod p$ \\
\midrule
1 & $p - 1$ \\
2 & $1$ \\
3 & $(p-1) \cdot 2^{-1} \bmod p$ \\
4 & $6^{-1} \bmod p$ \\
5 & $(p-1) \cdot 24^{-1} \bmod p$ \\
\bottomrule
\end{tabular}
\end{center}
\end{lemma}

\begin{proof}
We use the identity $(p-k)! = (p-1)! \big/ \prod_{j=p-k+1}^{p-1} j$ modulo $p$, combined with Wilson's theorem.

\textbf{$k=1$:} $(p-1)! \equiv -1 \pmod{p}$, so $(p-1)! \bmod p = p - 1$.

\textbf{$k=2$:} $(p-2)! = (p-1)!/(p-1) \equiv (-1)/(-1) = 1 \pmod{p}$.

\textbf{$k=3$:} $(p-3)! = (p-2)!/(p-2) \equiv 1/(-2) = -(2^{-1}) \equiv (p-1) \cdot 2^{-1} \pmod{p}$.

\textbf{$k=4$:} $(p-4)! = (p-3)!/(p-3) \equiv (-2^{-1})/(-3) = (6)^{-1} \pmod{p}$.

\textbf{$k=5$:} $(p-5)! = (p-4)!/(p-4) \equiv 6^{-1}/(-4) = -(24)^{-1} \equiv (p-1) \cdot 24^{-1} \pmod{p}$.
\end{proof}

\begin{theorem}[Closed Form for $S(p)$]
For any prime $p \ge 5$:
\[
S(p) \equiv -\frac{3}{8} \pmod{p},
\]
i.e., $S(p) = \bigl(p - 3 \cdot 8^{-1}\bigr) \bmod p$.
\end{theorem}

\begin{proof}
Summing the five residues:
\[
S(p) \equiv -1 + 1 - 2^{-1} + 6^{-1} - 24^{-1} \pmod{p}.
\]
Multiplying through by 24:
\[
24 \cdot S(p) \equiv -24 + 24 - 12 + 4 - 1 = -9 \pmod{p}.
\]
Hence $S(p) \equiv -9 \cdot 24^{-1} = -3 \cdot 8^{-1} \pmod{p}$.
\end{proof}

\section{Algorithm}

\begin{verbatim}
function solve():
    primes = sieve_of_eratosthenes(10^8)
    total = 0
    for p in primes where p >= 5:
        inv8 = power_mod(8, p - 2, p)
        sp = (p - 3 * inv8 % p) % p
        total += sp
    return total
\end{verbatim}

\textbf{Optimization:} Since $8 = 2^3$, compute $2^{-1} = (p+1)/2$ and cube it to get $8^{-1}$ in $O(1)$.

\section{Complexity Analysis}

\begin{itemize}
\item \textbf{Time:} $O(N \log \log N)$ for the sieve, plus $O(N / \ln N)$ for iterating over primes with $O(1)$ work each. Total: $O(N \log \log N)$ where $N = 10^8$.
\item \textbf{Space:} $O(N)$ for the prime sieve.
\end{itemize}
