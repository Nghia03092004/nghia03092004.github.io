\section{Problem Statement}
Find the pancake number $P(n)$: maximum prefix-reversal distance in $S_n$.

\section{Key Results}

\begin{theorem}[Breakpoint Bound]
$d(\pi) \ge \lceil b(\pi)/2 \rceil$ where $b(\pi)$ counts breakpoints.
\end{theorem}

\begin{theorem}[Bounds]
$\frac{15n}{14} \le P(n) \le \frac{5(n+5)}{3}$.
\end{theorem}
