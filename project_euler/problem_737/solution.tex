\section{Problem Statement}
Stack identical coins around a perpendicular line. The $n$-th coin rotates $\theta_n$ degrees. Find coins needed for 2020 complete loops (cumulative $\theta \ge 360 \times 2020$).

\section{Geometric Analysis}
\begin{theorem}
The angular displacement satisfies $\theta_n = 2\arcsin\!\left(\frac{1}{2\sqrt{n}}\right) + O(n^{-3/2})$, giving cumulative angle $\sim C\sqrt{n}$ for a constant $C$ determined by the exact geometry.
\end{theorem}
