\section{Problem Statement}
A telephone system has $10^6$ users. Call records are generated by a Lagged Fibonacci Generator:
\begin{align}
S_k &= [100003 - 200003k + 300007k^3] \bmod 10^6, \quad 1 \le k \le 55, \\
S_k &= [S_{k-24} + S_{k-55}] \bmod 10^6, \quad k > 55.
\end{align}
Call $n$ connects $S_{2n-1}$ to $S_{2n}$ (unless equal). Find the number of successful calls until 99\% of subscribers are in the same component as subscriber 524287.

\section{Mathematical Foundation}

\textbf{Theorem 1 (Tarjan, 1975).} A Disjoint Set Union with path compression and union by rank supports $m$ \textsc{Find} and $n$ \textsc{Union} operations in $O((m+n)\,\alpha(n))$ time, where $\alpha$ is the inverse Ackermann function.

\textbf{Proof.} Path compression flattens trees during \textsc{Find}, and union by rank bounds tree depth at $O(\log n)$ before compression. The amortized analysis via Ackermann functions (Tarjan, 1975) yields $O(\alpha(n))$ per operation. $\square$

\medskip

\textbf{Theorem 2.} The LFG with the stated initial conditions produces a deterministic pseudo-random sequence in $\{0, \ldots, 999999\}$.

\textbf{Proof.} The 55 initial values are computed by a cubic polynomial mod $10^6$. The recurrence $S_k = (S_{k-24} + S_{k-55}) \bmod 10^6$ is a linear additive recurrence, deterministic given the initial state. $\square$

\medskip

\textbf{Lemma 1 (Erd\H{o}s--R\'enyi Scaling).} In $G(n, M)$, a component of size $\geq (1-\varepsilon)n$ emerges at $M = \Theta(n \log(1/\varepsilon))$.

\textbf{Proof.} The giant component threshold is $M \sim n/2$. Above threshold, absorbing all but $\varepsilon n$ vertices requires $\Theta(n \log(1/\varepsilon))$ edges (Bollob\'as, \emph{Random Graphs}, 2001). For $n = 10^6$, $\varepsilon = 0.01$, this is consistent with the answer of 2,325,629. $\square$

\section{Algorithm}

\begin{algorithmic}[1]
\Function{FindCalls}{}
    \State Initialize DSU with $10^6$ elements
    \State Compute $S[1..55]$ from initial formula
    \State $\mathrm{calls} \gets 0$
    \While{$\mathrm{size}(\textsc{Find}(524287)) < 990000$}
        \State $\mathrm{caller} \gets \textsc{NextS}()$; $\mathrm{called} \gets \textsc{NextS}()$
        \If{$\mathrm{caller} \neq \mathrm{called}$}
            \State $\mathrm{calls} \gets \mathrm{calls} + 1$
            \State $\textsc{Union}(\mathrm{caller}, \mathrm{called})$
        \EndIf
    \EndWhile
    \State \Return calls
\EndFunction
\end{algorithmic}

\section{Complexity Analysis}

\begin{itemize}
    \item \textbf{Time:} $O(M \cdot \alpha(10^6))$ where $M \approx 2.3 \times 10^6$. Effectively $O(M)$.
    \item \textbf{Space:} $O(10^6)$ for DSU arrays and LFG buffer.
\end{itemize}
