\section{Problem Statement}

Species $S_{k,m}$ has $k$ bacterial types. Lifecycle rules from pseudo-random sequence $r_n$ ($r_0 = 306$, $r_{n+1} = r_n^2 \bmod 10007$). Find extinction probability $P_{500,10}$.

\section{Multi-Type Branching Process}

\begin{theorem}[Extinction Probability]
The extinction probability vector $\mathbf{q} = (q_0, \ldots, q_{k-1})$ is the smallest fixed point in $[0,1]^k$ of:
\[
q_i = \frac{1}{m} \sum_{j=0}^{m-1} g_{ij}(\mathbf{q})
\]
where $g_{ij}$ is the PGF for rule~$j$ of type~$i$.
\end{theorem}

\begin{proof}
Monotone iteration from $\mathbf{q}^{(0)} = \mathbf{0}$ converges to the smallest fixed point by the monotone convergence theorem applied to probability generating functions.
\end{proof}
