\section{Problem Statement}
Compute the expected number of keys to open all $n$ boxes (coupon collector variant).

\section{Coupon Collector}

\begin{theorem}
$E[T] = n H_n = n\sum_{k=1}^{n} \frac{1}{k}$.
\end{theorem}

\begin{proof}
Phase~$i$ (going from $i{-}1$ to $i$ distinct items) is geometric with parameter $(n-i+1)/n$, so expected length $n/(n-i+1)$. Summing gives $nH_n$.
\end{proof}

\begin{theorem}
$\mathrm{Var}(T) = n^2 \sum_{k=1}^n \frac{1}{k^2} - nH_n \sim \frac{\pi^2 n^2}{6}$.
\end{theorem}

\section{Asymptotics}
$E[T] = n\ln n + \gamma n + \frac{1}{2} + O(1/n)$ where $\gamma \approx 0.5772$.
