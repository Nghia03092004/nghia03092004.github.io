\section{Problem Statement}

An array $A_n[0..n-1]$ of zeros undergoes pseudo-random point updates. After step~$i$, $M_n(i) = \max_{0\le p\le q<n}\sum_{j=p}^{q}A_n[j]$. Compute $S(n,l)=\sum_{i=1}^{l}M_n(i)$.

Find $S(10\,000\,003,\,10\,200\,000) - S(10\,000\,003,\,10\,000\,000)$.

\section{Mathematical Foundation}

\begin{theorem}[Kadane, 1984]
The maximum subarray sum satisfies
\[
M = \max\!\Bigl(0,\;\max_{1\le q\le n}\bigl(P_q - \min_{0\le p<q}P_p\bigr)\Bigr)
\]
where $P_k = \sum_{j=0}^{k-1}A[j]$ and $P_0=0$.
\end{theorem}

\begin{proof}
Every contiguous sum $\sum_{j=p}^{q}A[j] = P_{q+1}-P_p$. Maximizing over $0\le p\le q<n$ is equivalent to maximizing $P_b - P_a$ for $0\le a<b\le n$. Including the empty subarray (sum~$0$) gives the formula.
\end{proof}

\begin{theorem}[Segment Tree Maintenance]
\label{thm:segtree}
A segment tree on prefix sums $P_0,\ldots,P_n$ supports point updates to~$A$ in $O(\log n)$ and global max-subarray queries in $O(1)$. Each node for $[l,r]$ stores $(\max P,\,\min P,\,\mathrm{max\_sub},\,\mathrm{total})$.
\end{theorem}

\begin{proof}
For node~$v$ with children $L=[l,m]$ and $R=[m{+}1,r]$, the maximum subarray sum is
\[
\mathrm{max\_sub}(v) = \max\bigl(\mathrm{max\_sub}(L),\;\mathrm{max\_sub}(R),\;\mathrm{max\_prefix}(R)-\mathrm{min\_prefix}(L)\bigr).
\]
The three cases correspond to: (1)~both endpoints in $L$, (2)~both in $R$, (3)~left endpoint in $L$ and right in $R$. The merge is correct by exhaustion of cases. Updating $A[k]$ by~$\delta$ is a suffix update on $P_{k+1},\ldots,P_n$, handled via lazy propagation in $O(\log n)$.
\end{proof}

\begin{lemma}[Non-Negativity]
$M_n(i)\ge 0$ always, since the empty subarray has sum~$0$.
\end{lemma}

\begin{proof}
Immediate from the definition.
\end{proof}

\section{Algorithm}

Build a segment tree on prefix sums. For each update~$i$: apply the suffix update to the tree; query the root's $\mathrm{max\_sub}$; accumulate.

\section{Complexity Analysis}

\begin{itemize}
\item \textbf{Time:} $O(n + l\log n)$. For $n\approx 10^7$, $l=2\times 10^5$: $\sim 4.6\times 10^6$ operations.
\item \textbf{Space:} $O(n)$ for the segment tree.
\end{itemize}
