\section{Problem Statement}

Sam and Max have digital clocks with 7-segment displays. Starting from a number, they repeatedly compute the digit sum until a single digit remains. Sam's clock turns off all segments then turns on the new display at each step. Max's clock only toggles differing segments. For all primes $p$ with $10^7 < p < 2 \times 10^7$, compute:
\[
    \sum_p \bigl(\text{Sam}(p) - \text{Max}(p)\bigr)
\]

\section{Mathematical Foundation}

\begin{definition}[Segment Encoding]
Each digit $d \in \{0,\ldots,9\}$ is a 7-bit mask $M(d)$. For a number $n$ with digits $d_1\cdots d_k$:
\[
    \text{seg}(n) = \sum_{i=1}^{k} |M(d_i)|
\]
Note: digit 7 uses 4 segments (including top-left vertical $f$), per the Project Euler specification.
\end{definition}

\begin{lemma}[Sam's Cost]
For the chain $n_0, n_1, \ldots, n_K$:
\[
    \text{Sam}(p) = 2\sum_{i=0}^{K} \text{seg}(n_i)
\]
\end{lemma}

\begin{proof}
Sam turns on all segments of $n_i$ (cost $\text{seg}(n_i)$) then turns them off (cost $\text{seg}(n_i)$), for each step $i = 0, \ldots, K$.
\end{proof}

\begin{lemma}[Max's Cost]
\[
    \text{Max}(p) = \text{seg}(n_0) + \sum_{i=0}^{K-1}\text{ham}(n_i, n_{i+1}) + \text{seg}(n_K)
\]
where $\text{ham}(n_i, n_{i+1})$ is the Hamming distance of the composite bitmasks (right-aligned, left-padded with zero).
\end{lemma}

\begin{proof}
Max turns on $n_0$ from blank, toggles only differing segments at each transition, and turns off $n_K$ to blank.
\end{proof}

\begin{theorem}[Difference Identity]
For bitmasks $A, B$: $|A| + |B| = |A \oplus B| + 2|A \wedge B|$. Consequently:
\[
    \text{Sam}(p) - \text{Max}(p) = 2\sum_{i=0}^{K-1} \text{common}(n_i, n_{i+1})
\]
where $\text{common}(n_i, n_{i+1})$ counts shared segments between consecutive displays.
\end{theorem}

\begin{proof}
For bitmasks $A, B \subseteq \{0,1\}^7$: each bit is in $A \cap B$, $A \setminus B$, or $B \setminus A$. Hence $|A| + |B| = |A \oplus B| + 2|A \wedge B|$.

Expanding the difference and using $\text{seg}(n_i) + \text{seg}(n_{i+1}) = \text{ham}(n_i,n_{i+1}) + 2\,\text{common}(n_i,n_{i+1})$, telescoping yields:
\[
    \text{Sam} - \text{Max} = 2\sum_{i=0}^{K-1}\text{common}(n_i, n_{i+1}). \qedhere
\]
\end{proof}

\section{Algorithm}

\begin{enumerate}
    \item Sieve primes in $(10^7, 2\times 10^7)$.
    \item Precompute $\text{seg\_count}[d]$ and $\text{bitmask}[d]$ for $d = 0, \ldots, 9$.
    \item For each prime $p$:
    \begin{enumerate}
        \item Build the digital root chain: $n_0 = p$, $n_{i+1} = \text{digit\_sum}(n_i)$.
        \item For each consecutive pair: compute bitwise AND of right-aligned digit bitmasks, sum the popcount.
        \item Add $2 \times \text{total\_common}$ to the global sum.
    \end{enumerate}
\end{enumerate}

\section{Complexity Analysis}

\begin{itemize}
    \item \textbf{Time:} $O(N \log \log N)$ for the sieve ($N = 2 \times 10^7$); $O(\log p)$ per prime for the chain. Total: $O(N \log \log N)$.
    \item \textbf{Space:} $O(N)$ for the sieve.
\end{itemize}
