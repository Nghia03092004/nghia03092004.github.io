\section{Problem}
Find $\sum_{m=2}^{1000} \pi(m)$ where $\pi(m)$ is the Pisano period.
\begin{theorem}
$\pi(m)$ exists with $\pi(m) \le 6m$. For coprime $a,b$: $\pi(ab) = \mathrm{lcm}(\pi(a), \pi(b))$.
\end{theorem}
