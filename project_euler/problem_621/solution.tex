\section{Problem Statement}

Let $T(n)$ denote the number of ordered triples $(a,b,c)$ of non-negative integers such that
\[
\frac{a(a+1)}{2} + \frac{b(b+1)}{2} + \frac{c(c+1)}{2} = n.
\]
Find $\sum_i T(n_i)$ for specified values $n_i$.

\section{Mathematical Foundation}

\begin{theorem}[Reduction to Sums of Odd Squares]
\label{thm:reduction}
For every non-negative integer~$n$, $T(n)$ equals the number of ordered triples $(x,y,z)$ of odd positive integers satisfying $x^2 + y^2 + z^2 = 8n+3$.
\end{theorem}

\begin{proof}
The $k$-th triangular number satisfies $8t_k + 1 = 4k^2 + 4k + 1 = (2k+1)^2$. The map $k \mapsto 2k+1$ bijects $\mathbb{Z}_{\ge 0}$ onto the odd positive integers. Given $t_a + t_b + t_c = n$, summing gives
\[
(2a+1)^2 + (2b+1)^2 + (2c+1)^2 = 8n + 3.
\]
Conversely, if $x^2+y^2+z^2 = 8n+3$ with $x,y,z$ odd and positive, then $x=2a+1$, $y=2b+1$, $z=2c+1$ for unique $a,b,c \ge 0$, and $t_a+t_b+t_c = n$. The correspondence preserves ordering.
\end{proof}

\begin{theorem}[Gauss's Eureka Theorem]
\label{thm:gauss}
Every non-negative integer is the sum of three triangular numbers, i.e., $T(n) \ge 1$ for all $n \ge 0$.
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:reduction}, $T(n) \ge 1$ iff $8n+3$ is a sum of three odd positive squares. By Legendre's three-square theorem, $m$ fails to be a sum of three squares iff $m = 4^a(8b+7)$. Since $8n+3 \equiv 3 \pmod{8}$, it is never of this form, so $8n+3 = x^2+y^2+z^2$ for some integers $x,y,z$. Moreover, $8n+3 \equiv 3 \pmod{8}$ forces all three squares to be odd: an even square is $\equiv 0 \pmod{4}$, and one cannot express $3 \pmod{8}$ as a sum of three values each in $\{0,1\} \pmod{4}$ unless all three are~$1$.
\end{proof}

\begin{lemma}[Class Number Formula]
\label{lem:class}
For positive integer~$m$, the number of representations $r_3(m) = |\{(x,y,z) \in \mathbb{Z}^3 : x^2+y^2+z^2 = m\}|$ satisfies $r_3(m) = 12\,H(m)$, where $H(m)$ is the Hurwitz class number.
\end{lemma}

\begin{proof}
Classical result of Gauss--Dirichlet; see Grosswald, \textit{Representations of Integers as Sums of Squares}, Chapter~8. The count $r_3(m)$ relates to the class number of the imaginary quadratic order of discriminant $-4m$.
\end{proof}

\begin{theorem}[Extraction of $T(n)$]
Since $8n+3 \equiv 3 \pmod{8}$, every representation $(x,y,z) \in \mathbb{Z}^3$ with $x^2+y^2+z^2 = 8n+3$ has all three entries odd. Thus $r_3(8n+3)$ counts signed odd triples, and the count of ordered triples of positive odd integers is $r_3(8n+3)/8$ (accounting for the $2^3$ sign choices per triple in the generic case).
\end{theorem}

\begin{proof}
Each positive triple $\{|x|,|y|,|z|\}$ with all entries distinct yields $2^3 = 8$ signed triples $(\pm x, \pm y, \pm z)$. Since $8n+3 \equiv 3 \pmod{4}$, no two of $x^2, y^2, z^2$ can be equal when $m = 3 \pmod{4}$ except in degenerate cases that require separate accounting. In the generic case, $T(n) = r_3(8n+3)/8$.
\end{proof}

\section{Algorithm}

\begin{enumerate}
\item Compute $m = 8n + 3$.
\item Factorize $m$ and compute $H(m)$ via the Hurwitz class number formula.
\item Obtain $r_3(m) = 12\,H(m)$ and extract $T(n)$.
\item For brute force: iterate over $a$ with $t_a \le n$, then $b$ with $t_b \le n - t_a$, and check whether $n - t_a - t_b$ is triangular.
\end{enumerate}

\begin{verbatim}
function brute_force_T(n):
    count = 0
    for a = 0 to floor((sqrt(8n+1)-1)/2):
        rem = n - a*(a+1)/2
        for b = 0 to floor((sqrt(8*rem+1)-1)/2):
            rem2 = rem - b*(b+1)/2
            disc = 8*rem2 + 1
            s = isqrt(disc)
            if s*s == disc and s % 2 == 1:
                count += 1
    return count
\end{verbatim}

\section{Complexity Analysis}

\begin{itemize}
\item \textbf{Time:} Brute force runs in $O(n)$. The analytic method via class numbers runs in $O(n^{1/2+\varepsilon})$, dominated by factoring $8n+3$.
\item \textbf{Space:} $O(1)$ auxiliary for brute force; $O(n^{1/2})$ for sieve-based factorization.
\end{itemize}
