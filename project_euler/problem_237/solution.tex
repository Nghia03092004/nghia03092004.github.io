\section{Problem Statement}
Let $T(n)$ be the number of tours on a $4 \times n$ board that visit every square exactly once, start at the top-left corner, end at the bottom-left corner, moving by single steps in cardinal directions. Given $T(10) = 2329$, find $T(10^{12}) \bmod 10^8$.

\section{Mathematical Foundation}

\begin{theorem}[Transfer Matrix Representation]
The count $T(n)$ can be expressed as
\[
T(n) = \mathbf{e}_{\mathrm{start}}^\top \cdot \mathbf{M}^{n-1} \cdot \mathbf{e}_{\mathrm{end}}
\]
where $\mathbf{M}$ is a $k \times k$ transfer matrix encoding valid column-to-column transitions of boundary connectivity states.
\end{theorem}

\begin{proof}
Decompose the grid into columns. At each column boundary, the partial Hamiltonian path induces a connectivity pattern among the 4 row positions. This connectivity forms a finite state space. The transfer matrix $M_{ij}$ counts valid single-column fillings transitioning state~$i$ to state~$j$. The product $\mathbf{M}^{n-1}$ accounts for all $n-1$ column transitions.
\end{proof}

\begin{lemma}[Linear Recurrence]
$T(n)$ satisfies a linear recurrence of order $k$ with characteristic polynomial $\det(xI - \mathbf{M})$.
\end{lemma}

\begin{proof}
By the Cayley--Hamilton theorem, $\mathbf{M}$ satisfies its characteristic polynomial $p(\mathbf{M}) = 0$. Since $T(n)$ is a linear function of $\mathbf{M}^{n-1}$, it inherits the recurrence defined by the coefficients of~$p(x)$.
\end{proof}

\begin{theorem}[Matrix Exponentiation]
For any integer $m$ and modulus $M$, the matrix power $\mathbf{M}^m \bmod M$ is computed in $O(k^3 \log m)$ arithmetic operations via repeated squaring.
\end{theorem}

\begin{proof}
Express $m$ in binary: $m = \sum_j b_j 2^j$. Then $\mathbf{M}^m = \prod_{j:\, b_j=1} \mathbf{M}^{2^j}$. Each squaring costs $O(k^3)$ multiplications with $\lfloor\log_2 m\rfloor + 1$ squarings total. Modular reduction preserves correctness since $(AB)\bmod M = ((A\bmod M)(B\bmod M))\bmod M$ entry-wise.
\end{proof}

For the $4 \times n$ board, enumeration yields $k \approx 6$ boundary states. The recurrence is verified against $T(10) = 2329$.

\section{Algorithm}

\begin{verbatim}
function solve():
    M = construct_transfer_matrix()       // k x k, k ~ 6
    R = matrix_power(M, 10^12 - 1, 10^8) // repeated squaring mod 10^8
    return e_start^T * R * e_end mod 10^8
\end{verbatim}

\section{Complexity Analysis}
\begin{itemize}
    \item \textbf{Time:} $O(k^3 \log n)$ where $k \approx 6$ and $n = 10^{12}$, yielding $\sim 8640$ operations.
    \item \textbf{Space:} $O(k^2)$ for the matrix.
\end{itemize}
