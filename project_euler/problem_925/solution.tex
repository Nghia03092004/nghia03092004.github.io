\section{Problem Statement}

The continued fraction expansion of $\sqrt{2}$ is $[1; 2, 2, 2, \ldots]$. Let $p_n/q_n$ denote the $n$-th convergent. Find $p_{100} + q_{100} \bmod 10^9+7$.

\section{Mathematical Foundation}

\begin{theorem}[Convergent Recurrence]
Let $[a_0; a_1, a_2, \ldots]$ be a simple continued fraction with $p_{-1}=1$, $p_0=a_0$, $q_{-1}=0$, $q_0=1$, and for $n \geq 1$:
\[
p_n = a_n p_{n-1} + p_{n-2}, \qquad q_n = a_n q_{n-1} + q_{n-2}.
\]
Then $p_n/q_n$ is the $n$-th convergent.
\end{theorem}

\begin{proof}
By induction on the matrix identity
\[
\begin{pmatrix} p_n & p_{n-1} \\ q_n & q_{n-1} \end{pmatrix} = \prod_{i=0}^{n} \begin{pmatrix} a_i & 1 \\ 1 & 0 \end{pmatrix}. \tag{M}
\]
Base case ($n=0$): the right side is $\bigl(\begin{smallmatrix} a_0 & 1 \\ 1 & 0 \end{smallmatrix}\bigr)$, matching $(p_0,p_{-1};q_0,q_{-1})$. Inductive step: multiplying on the right by $\bigl(\begin{smallmatrix} a_n & 1 \\ 1 & 0 \end{smallmatrix}\bigr)$ yields the recurrence.
\end{proof}

\begin{theorem}[Determinant Identity / Pell Connection]
For all $n \geq 0$: $p_n q_{n-1} - p_{n-1}q_n = (-1)^{n+1}$. For $\sqrt{2}$: $p_n^2 - 2q_n^2 = (-1)^{n+1}$.
\end{theorem}

\begin{proof}
Taking determinants of~(M): $\det = \prod_{i=0}^n \det\bigl(\begin{smallmatrix} a_i & 1 \\ 1 & 0 \end{smallmatrix}\bigr) = (-1)^{n+1}$.
\end{proof}

\begin{lemma}[Specialization to $\sqrt{2}$]
For $\sqrt{2} = [1;2,2,2,\ldots]$: $a_0=1$, $a_n=2$ for $n \geq 1$, giving
\[
p_n = 2p_{n-1}+p_{n-2},\quad q_n = 2q_{n-1}+q_{n-2},
\]
with $p_0=1$, $p_1=3$, $q_0=1$, $q_1=2$.
\end{lemma}

\begin{proof}
Direct substitution: $p_1 = 2\cdot 1 + 1 = 3$, $q_1 = 2\cdot 1 + 0 = 2$.
\end{proof}

\begin{theorem}[Matrix Exponentiation]
For $n \geq 1$:
\[
\begin{pmatrix} p_n \\ p_{n-1}\end{pmatrix} = \begin{pmatrix} 2 & 1 \\ 1 & 0\end{pmatrix}^{n-1}\begin{pmatrix} 3 \\ 1\end{pmatrix},
\]
enabling $O(\log n)$ computation.
\end{theorem}

\begin{proof}
Rewrite $p_n = 2p_{n-1}+p_{n-2}$ in matrix form and iterate from $(p_1,p_0)^T = (3,1)^T$.
\end{proof}

\begin{lemma}[Growth Rate]
$q_n = \frac{(1+\sqrt{2})^{n+1} - (1-\sqrt{2})^{n+1}}{2\sqrt{2}}$. Both $p_n$ and $q_n$ grow as $\Theta\!\left((1+\sqrt{2})^n\right)$.
\end{lemma}

\begin{proof}
The characteristic equation $\lambda^2 - 2\lambda - 1 = 0$ has roots $1 \pm \sqrt{2}$. Solving with $q_0=1$, $q_1=2$ yields the closed form. Since $|1-\sqrt{2}|<1$, the subdominant term vanishes.
\end{proof}

\section{Algorithm}

Iterate the recurrence $p_n = 2p_{n-1}+p_{n-2}$, $q_n = 2q_{n-1}+q_{n-2}$ from $n=2$ to $n=100$, reducing modulo $10^9+7$ at each step. Return $(p_{100}+q_{100}) \bmod (10^9+7)$.

\section{Complexity Analysis}

\begin{itemize}
    \item \textbf{Time:} $O(n)$ iteratively; $O(\log n)$ via matrix exponentiation.
    \item \textbf{Space:} $O(1)$.
\end{itemize}
