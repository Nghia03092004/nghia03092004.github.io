\section{Problem Statement}
A ``castle'' on a $w \times h$ grid is a connected shape with a battlement profile: column heights form a sequence $(h_1, \ldots, h_w)$ with $1 \le h_i \le h$, no two adjacent columns the same height, the bottom row fully filled, and at least one column reaching height~$h$. Compute the required combination of $F(w,h)$ values modulo $10^9 + 7$.

\section{Mathematical Foundation}

\begin{theorem}[Battlement Count]
The number of sequences $(h_1, \ldots, h_w)$ with $1 \le h_i \le h$ and $h_i \ne h_{i+1}$ is
\[
B(w, h) = h \cdot (h-1)^{w-1}.
\]
\end{theorem}

\begin{proof}
The first column has $h$ choices. Each subsequent column has $h - 1$ choices (any height except the previous). By the multiplication principle, $B(w,h) = h(h-1)^{w-1}$.
\end{proof}

\begin{theorem}[Inclusion--Exclusion for Maximum Height]
The number of battlement profiles with at least one column of height~$h$ is
\[
F_{\mathrm{simple}}(w, h) = h(h-1)^{w-1} - (h-1)(h-2)^{w-1}.
\]
\end{theorem}

\begin{proof}
Let $A$ be the set of all battlement profiles in $\{1,\ldots,h\}$ and $A'$ the subset restricted to $\{1,\ldots,h-1\}$. Then $|A \setminus A'| = B(w,h) - B(w,h-1)$.
\end{proof}

\begin{lemma}[Transfer-Matrix Method]
For structural constraints beyond simple battlement profiles, define a transfer matrix $T$ over valid column profiles. Then
\[
F(w, h) = \mathbf{1}^T \cdot T^{w-1} \cdot \mathbf{s}
\]
where $\mathbf{s}$ encodes valid starting profiles, with inclusion--exclusion to enforce the maximum-height constraint.
\end{lemma}

\begin{proof}
The transfer matrix encodes all valid local adjacencies between consecutive columns. The $(w-1)$-th power counts valid length-$w$ sequences. Inclusion--exclusion removes profiles that never reach height~$h$.
\end{proof}

\section{Algorithm}
\begin{verbatim}
1. Simple model: F = h*(h-1)^(w-1) - (h-1)*(h-2)^(w-1) mod p
2. Transfer-matrix model (if needed):
   a. Enumerate valid column profiles
   b. Build transfer matrix T
   c. Compute T^(w-1) via matrix exponentiation mod p
   d. Sum over valid start/end states
   e. Subtract profiles never reaching height h
\end{verbatim}

\section{Complexity Analysis}
\begin{itemize}
    \item \textbf{Time (simple model):} $O(\log w)$ via modular exponentiation.
    \item \textbf{Time (transfer-matrix):} $O(h^3 \cdot 2^h \cdot \log w)$ with full profile states.
    \item \textbf{Space:} $O(h^2)$ for the transfer matrix, or $O(4^h)$ for full profile DP.
\end{itemize}
