\section{Problem Statement}

A \textbf{sqube} is $n = p^2 q^3$ where $p, q$ are distinct primes. A number is \textbf{prime-proof} if replacing any single digit with any other digit never yields a prime. Find the 200th smallest prime-proof sqube whose decimal representation contains ``200''.

\section{Mathematical Foundation}

\begin{theorem}[Sqube Characterization]
A positive integer $n$ is a sqube iff $n = p^2 q^3$ for distinct primes $p, q$. The form $p^3 q^2$ is the same as $q^2 p^3$ with roles swapped.
\end{theorem}

\begin{proof}
By definition. The prime factorization uniquely determines $p$ and $q$. \qed
\end{proof}

\begin{theorem}[Deterministic Miller--Rabin]
For $n < 3.2 \times 10^{14}$, the Miller--Rabin test with witnesses $\{2, 3, 5, 7, 11, 13\}$ is deterministic.
\end{theorem}

\begin{proof}
Jaeschke (1993) proved no strong pseudoprimes to these bases exist below $3.2 \times 10^{14}$. Our candidates and their digit substitutions are below $10^{13}$, within this bound. \qed
\end{proof}

\begin{lemma}[Prime-Proof Condition]
$n$ with digits $d_1 \cdots d_m$ is prime-proof iff for every position $i$ and every $d' \neq d_i$ (with $d' \neq 0$ when $i = 1$), the modified number is composite.
\end{lemma}

\begin{proof}
Direct from the definition of single-digit substitution. \qed
\end{proof}

\begin{lemma}[Substitution Count]
A number with $m$ digits has at most $9m - 1$ single-digit substitutions: 8 for the leading digit, 9 for each subsequent digit.
\end{lemma}

\begin{proof}
Position 1: $|\{1,\ldots,9\} \setminus \{d_1\}| = 8$. Position $i > 1$: $|\{0,\ldots,9\} \setminus \{d_i\}| = 9$. Total: $8 + 9(m-1) = 9m - 1$. \qed
\end{proof}

\section{Algorithm}

\begin{verbatim}
1. Generate all squbes p^2*q^3 below bound B
   (iterate q over primes <= B^{1/3},
    p over primes <= sqrt(B/q^3), p != q)
2. Filter: keep those containing "200" as substring
3. Sort candidates
4. For each in order, check prime-proof:
   - For each digit position and replacement:
     test primality via Miller-Rabin({2,3,5,7,11,13})
   - If any substitution is prime: not prime-proof
5. Return the 200th prime-proof sqube
\end{verbatim}

\section{Complexity Analysis}

\begin{itemize}
    \item \textbf{Time:} Sqube generation: $O(B^{1/2}/\log B)$ pairs. Sorting: $O(S \log S)$ candidates. Prime-proof check: $O(m \cdot \log^2 n)$ per candidate ($m$ = number of digits).
    \item \textbf{Space:} $O(S)$ to store filtered squbes.
\end{itemize}
