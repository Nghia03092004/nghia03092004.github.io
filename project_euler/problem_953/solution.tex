\section{Problem Statement}

A sorting network on~$n$ wires consists of comparators $(i,j)$ that swap wires~$i$ and~$j$ if they are out of order. By the 0-1 principle, a network sorts all inputs iff it sorts all $2^n$ binary inputs. For $n=16$, an optimal sorting network has 60 comparators. Given the AKS network construction, find the number of binary inputs where at least one intermediate state has exactly 8 ones and 8 zeros in sorted order.

\section{Mathematical Foundation}

\begin{theorem}[Zero-One Principle]
A comparator network sorts all inputs from a totally ordered set if and only if it sorts all $2^n$ binary input sequences.
\end{theorem}

\begin{proof}
The forward direction is trivial. For the converse, suppose the network fails to sort some input $x_1, \ldots, x_n$, producing outputs with $y_i > y_j$ for some $i < j$. Define $b_k = [x_k > y_j]$. Since comparators commute with monotone functions, the network applied to $(b_1, \ldots, b_n)$ produces $b'_k = [y_k > y_j]$. Then $b'_i = 1 > 0 = b'_j$ with $i < j$, so this binary input is not sorted---a contradiction.
\end{proof}

\begin{lemma}[State Determinism]
For a fixed input and a given comparator network, the state after each comparator is uniquely determined.
\end{lemma}

\begin{proof}
Each comparator applies a deterministic $\min/\max$ operation. Starting from a fixed input, repeated application of deterministic operations yields a unique sequence of intermediate states.
\end{proof}

\begin{theorem}[Balanced State Count]
Let $\mathcal{N}$ be a correct sorting network on $n = 16$ wires. The number of binary inputs that pass through the balanced sorted state $(0^8, 1^8)$ at some intermediate stage equals $\binom{16}{8} = 12870$.
\end{theorem}

\begin{proof}
A comparator network preserves the number of ones in a binary input (each comparator merely rearranges values). Thus only the $\binom{16}{8}$ inputs with exactly 8 ones can possibly reach the state $(0^8, 1^8)$. Conversely, every such input \emph{must} reach this state at the final output, since $\mathcal{N}$ correctly sorts all binary inputs. Therefore the count is exactly $\binom{16}{8} = 12870$.
\end{proof}

\section{Algorithm}

\begin{verbatim}
function CountBalancedInputs(network, n):
    target = (0,...,0,1,...,1)   // 8 zeros then 8 ones
    count = 0
    for each binary input b in {0,1}^n:
        state = b
        for each comparator (i,j) in network:
            if state[i] > state[j]: swap(state[i], state[j])
            if state == target: count++; break
    return count
\end{verbatim}

Alternatively, by the theorem above, the answer is simply $\binom{16}{8}$.

\section{Complexity Analysis}

\begin{itemize}
\item \textbf{Time (brute force):} $O(2^n \cdot C)$ where $C = 60$. For $n=16$: $\approx 4 \times 10^6$ operations.
\item \textbf{Time (analytic):} $O(1)$ via the closed form $\binom{16}{8}$.
\item \textbf{Space:} $O(n)$ for the state vector.
\end{itemize}
