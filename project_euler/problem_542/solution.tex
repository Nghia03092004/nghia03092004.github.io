\section{Problem Statement}

A geometric progression (GP) consists of terms $a, ar, ar^2, \ldots, ar^{k-1}$ where all terms are distinct positive integers not exceeding~$n$. Define $G(n)$ as the maximum possible sum of such a GP. Compute $\sum_{n=4}^{N} G(n)$ for a given bound~$N$.

\section{Mathematical Foundation}

\begin{theorem}[Rational ratio characterization]
Every GP of distinct positive integers with ratio $r = p/q$ (in lowest terms, $\gcd(p,q)=1$, $p > q \ge 1$) and $k$ terms has the form $m \cdot q^{k-1-i} \cdot p^i$ for $i = 0, 1, \ldots, k-1$, where $m \ge 1$ is a positive integer.
\end{theorem}

\begin{proof}
The $i$-th term is $a \cdot r^i = a \cdot p^i / q^i$. For integrality, $q^{k-1} \mid a$. Writing $a = m \cdot q^{k-1}$, the $i$-th term becomes $m \cdot q^{k-1-i} \cdot p^i$, which is a positive integer. Distinctness follows from $p \neq q$ and $\gcd(p,q) = 1$.
\end{proof}

\begin{lemma}[Sum formula]
The sum of a GP with parameters $(m, p, q, k)$ is
\[
S = m \cdot q^{k-1} \cdot \frac{p^k - q^k}{p - q}.
\]
\end{lemma}

\begin{proof}
By the geometric series formula: $S = a \cdot (r^k - 1)/(r - 1)$ with $a = m q^{k-1}$ and $r = p/q$.
\end{proof}

\begin{theorem}[Bound constraint and optimization]
All terms are at most $n$ if and only if $m \cdot p^{k-1} \le n$. The sum $S$ is maximized by choosing $m = \lfloor n / p^{k-1} \rfloor$ and optimizing over all valid $(p, q, k)$ triples.
\end{theorem}

\begin{proof}
The largest term is $m \cdot p^{k-1}$. Since $S$ is linear in $m$, maximizing $m$ subject to $m \cdot p^{k-1} \le n$ maximizes $S$ for fixed $(p, q, k)$. The global maximum is found by enumerating all valid triples.
\end{proof}

\section{Algorithm}

\begin{enumerate}
\item For each $n$ from 4 to $N$:
\item Enumerate $k \ge 2$, $p \ge 2$ with $p^{k-1} \le n$, and $q < p$ with $\gcd(p,q)=1$.
\item Set $m = \lfloor n / p^{k-1} \rfloor$ and compute $S = m \cdot q^{k-1} \cdot (p^k - q^k)/(p-q)$.
\item Track the maximum $S$ over all $(p, q, k)$; this is $G(n)$.
\item Accumulate the sum $\sum G(n)$.
\end{enumerate}

\section{Complexity Analysis}

\begin{itemize}
\item \textbf{Time:} $O(N \sqrt{N} \log N)$ naively; $O(N^{2/3} \log^2 N)$ with breakpoint optimization.
\item \textbf{Space:} $O(1)$ with streaming summation, or $O(N)$ if storing all $G(n)$ values.
\end{itemize}
